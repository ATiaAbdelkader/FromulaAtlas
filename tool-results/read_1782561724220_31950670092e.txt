   463→    // Micro sulfatos + P (MAP/MKP u ortofosfato en NPK)
   464→    if (has('metal_sulfate') && (has('po4') || isMap || isMkp)) return 'I';
   465→
   466→    // Quelatos + fuerte P ortofosfato
   467→    if ((A.chelate || B.chelate) && (isMap || isMkp || (has('po4') && (A.npk || B.npk)))) return 'R';
   468→
   469→    // KNO3-family + sulfato amonio / SOP (salting)
   470→    if ((aid === 'nks' || bid === 'nks' || aid === 'nk_mg' || bid === 'nk_mg') && (has('nh4') && has('so4'))) return 'R';
   471→    if ((aid === 'nks' || bid === 'nks') && (aid === 'sop' || bid === 'sop')) return 'R';
   472→
   473→    // Ácidos fuertes mezclados entre sí o con bases implícitas
   474→    if (A.strong_acid && B.strong_acid && aid !== bid) return 'R';
   475→
   476→    return 'C';
   477→  }

   578→  function detailTemplate(aid, bid, level) {
   579→    if (aid === bid) return DETAIL_KEYS.same;
   580→
   581→    const A = traits(aid);
   582→    const B = traits(bid);
   583→    const ca = A.ca_no3 || B.ca_no3;
   584→    const mapMkp = aid === 'map' || bid === 'map' || aid === 'mkp' || bid === 'mkp';
   585→
   586→    if (level === 'I') {
   587→      if (mapMkp && ca && (A.po4 || B.po4)) return DETAIL_KEYS.ca_phosphate_mkp;
   588→      if (ca && (A.h3po4 || B.h3po4)) return DETAIL_KEYS.ca_acid_phosphoric;
   589→      if (ca && (A.h2so4 || B.h2so4)) return DETAIL_KEYS.ca_acid_sulfuric;
   590→      if (ca && (A.so4 || B.so4)) {
   591→        if ((A.nh4 && A.so4) || (B.nh4 && B.so4)) return DETAIL_KEYS.ca_sulfate;
   592→        if (aid === 'sop' || bid === 'sop' || aid === 'sulfato_magnesio' || bid === 'sulfato_magnesio' || aid === 'nks' || bid === 'nks') return DETAIL_KEYS.ca_sulfate;
   593→      }
   594→      if ((A.metal_sulfate || B.metal_sulfate) && (A.po4 || B.po4)) return DETAIL_KEYS.metal_sulfate_phosphate;
   595→      return DETAIL_KEYS.default_i;
   596→    }
   597→    if (level === 'R') {
   598→      if ((A.chelate || B.chelate) && (A.po4 || B.po4)) return DETAIL_KEYS.chelate_phosphate;
   599→      if ((aid === 'nks' || bid === 'nks' || aid === 'nk_mg' || bid === 'nk_mg') && ((A.nh4 && A.so4) || (B.nh4 && B.so4))) return DETAIL_KEYS.kno3_sulfate_salting;
   600→      if ((aid === 'sop' || bid === 'sop') && (aid === 'nks' || bid === 'nks')) return DETAIL_KEYS.kno3_sulfate_salting;
   601→      if (A.strong_acid && B.strong_acid) return DETAIL_KEYS.strong_acids;
   602→      if ((A.chelate || B.chelate) && (A.mg || B.mg) && (aid === 'nitrato_magnesio' || bid === 'nitrato_magnesio')) return DETAIL_KEYS.chelate_phosphate;
   603→      return DETAIL_KEYS.default_r;
   604→    }
   605→    return DETAIL_KEYS.default_c;
   606→  }

   669→    const n = FERTILIZERS.length;
   670→    const table = document.getElementById('compatTable');
   671→    const esc = function (s) {
   672→      return String(s).replace(/&/g, '&amp;').replace(/"/g, '&quot;').replace(/</g, '&lt;');
   673→    };
   674→    let thead = '<thead><tr><th class="corner" scope="col" title="Eje: filas = primera sustancia; columnas = segunda sustancia en el cruce."><span class="corner-title">Fertilizantes</span><span class="corner-sub">Mismo orden arriba y a la izquierda. Encabezados compactos (2 líneas máx.).</span></th>';
   675→    for (let j = 0; j < n; j++) {
   676→      const nm = FERTILIZERS[j].name;
   677→      thead += '<th scope="col" class="col-vlabel" data-j="' + j + '" title="' + esc(nm) + '"><span class="vlabel-inner">' + esc(nm) + '</span></th>';
   678→    }
   679→    thead += '</tr></thead>';
   680→
   681→    let tbody = '<tbody>';
   682→    for (let i = 0; i < n; i++) {
   683→      tbody += '<tr>';
   684→      tbody += '<th scope="row" class="row-h" data-i="' + i + '" title="' + esc(FERTILIZERS[i].name) + '"><span class="row-num">' + (i + 1) + '.</span> ' + esc(FERTILIZERS[i].name) + '</th>';
   685→      for (let j = 0; j < n; j++) {
   686→        if (j > i) {
   687→          tbody += '<td class="empty-cell"></td>';
   688→          continue;
   689→        }
   690→        const ai = FERTILIZERS[i].id;
   691→        const aj = FERTILIZERS[j].id;
   692→        const lev = inferLevel(ai, aj);
   693→        const L = lev === 'C' ? 'c' : lev === 'R' ? 'r' : 'i';
   694→        const sym = lev === 'C' ? 'C' : lev === 'R' ? 'R' : 'I';
   695→        const tipTxt = FERTILIZERS[i].name + ' ↔ ' + FERTILIZERS[j].name + ' — ' +
   696→          (sym === 'C' ? 'Compatible' : sym === 'R' ? 'Precaución' : 'Incompatible');
   697→        tbody += '<td><button type="button" class="cell-btn l-' + L + '" data-i="' + i + '" data-j="' + j + '" data-a="' + ai + '" data-b="' + aj + '" aria-label="' + esc(tipTxt) + '" title="' + esc(tipTxt) + '">' + sym + '</button></td>';
   698→      }
   699→      tbody += '</tr>';
   700→    }
   701→    tbody += '</tbody>';
   702→    table.innerHTML = thead + tbody;
   703→
   704→    table.querySelectorAll('thead th[data-j]').forEach(function (th) {
   705→      th.addEventListener('click', function () {
   706→        highlightFert(parseInt(th.getAttribute('data-j'), 10));
   707→      });
   708→    });
   709→    table.querySelectorAll('tbody th.row-h[data-i]').forEach(function (th) {
   710→      th.addEventListener('click', function () {
   711→        highlightFert(parseInt(th.getAttribute('data-i'), 10));
   712→      });
   713→    });
   714→    table.querySelectorAll('.cell-btn[data-a]').forEach(function (btn) {
   715→      btn.addEventListener('click', function (e) {
   716→        e.stopPropagation();
   717→        clearHighlights();
   718→        const i = parseInt(btn.getAttribute('data-i'), 10);
   719→        const j = parseInt(btn.getAttribute('data-j'), 10);
   720→        const a = btn.getAttribute('data-a');
   721→        const b = btn.getAttribute('data-b');
   722→        selectedPair = [i, j];
   723→        selectedFert = null;
   724→        btn.classList.add('selected');
   725→        const lev = inferLevel(a, b);
   726→        renderDetail(a, b, lev);
   727→        for (let ii = 0; ii < n; ii++) {
   728→          for (let jj = 0; jj <= ii; jj++) {
   729→            const b2 = document.querySelector('.cell-btn[data-i="' + ii + '"][data-j="' + jj + '"]');
   730→            if (!b2) continue;
   731→            if (ii === i && jj === j) {
   732→              b2.classList.add('selected', 'hl-both');
   733→            } else if (ii === i) {
   734→              b2.classList.add('hl-row');
   735→            } else if (jj === j) {
   736→              b2.classList.add('hl-col');
   737→            }
   738→          }
   739→        }
   740→        const rh = document.querySelector('.row-h[data-i="' + i + '"]');
   741→        const ch = document.querySelector('thead th[data-j="' + j + '"]');
