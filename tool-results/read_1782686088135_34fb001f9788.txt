   309→export function FreeToolsSection() {
   310→  const [search, setSearch] = useState('');
   311→  const [activeCategory, setActiveCategory] = useState<ToolCategory | 'All'>('All');
   312→  const [openTool, setOpenTool] = useState<ToolMeta | null>(null);
   313→  const [showHints, setShowHints] = useState(false);
   314→
   315→  // Comparison tray — holds up to 2 tool IDs for side-by-side comparison.
   316→  // Persists across tool-dialog opens/closes but not across page reloads (v1).
   317→  const [compareTray, setCompareTray] = useState<string[]>([]);
   318→  const [compareOpen, setCompareOpen] = useState(false);
   319→
   320→  // Ref to the search input — focused by the Ctrl+K / Cmd+K shortcut.
   321→  const searchInputRef = useRef<HTMLInputElement>(null);
   322→
   323→  // Favorites + Recently used (localStorage-persisted)
   324→  const FAV_KEY = 'nutriplant_tools_favorites_v1';
   325→  const REC_KEY = 'nutriplant_tools_recent_v1';
   326→  const [favorites, setFavorites] = useState<string[]>([]);
   327→  const [recent, setRecent] = useState<string[]>([]);
   328→
   329→  // Load persisted favorites + recent on mount; also handle #tool=<id> hash for shareable URLs
   330→  useEffect(() => {
   331→    try {
   332→      const fav = JSON.parse(localStorage.getItem(FAV_KEY) || '[]');
   333→      const rec = JSON.parse(localStorage.getItem(REC_KEY) || '[]');
   334→      if (Array.isArray(fav)) setFavorites(fav);
   335→      if (Array.isArray(rec)) setRecent(rec);
   336→    } catch { /* corrupt localStorage — ignore */ }
   337→    // Open tool from URL hash (#tool=<id>) on initial load — supports shared URLs
   338→    if (typeof window !== 'undefined') {
   339→      const m = window.location.hash.match(/#tool=([\w-]+)/);
   340→      if (m) {
   341→        const t = TOOLS.find(x => x.id === m[1]);
   342→        if (t) {
   343→          setOpenTool(t);
   344→          setRecent(prev => [t.id, ...prev.filter(id => id !== t.id)].slice(0, 5));
   345→        }
   346→      }
   347→    }
   348→    // eslint-disable-next-line react-hooks/exhaustive-deps
   349→  }, []);
   350→
   351→  // Persist favorites
   352→  useEffect(() => {
   353→    try { localStorage.setItem(FAV_KEY, JSON.stringify(favorites)); } catch { /* storage full or blocked */ }
   354→  }, [favorites]);
   355→
   356→  // Persist recent
   357→  useEffect(() => {
   358→    try { localStorage.setItem(REC_KEY, JSON.stringify(recent)); } catch { /* storage full or blocked */ }
   359→  }, [recent]);
   360→
   361→  const toggleFavorite = useCallback((id: string) => {
   362→    setFavorites(prev => prev.includes(id) ? prev.filter(x => x !== id) : [...prev, id]);
   363→  }, []);
   364→
   365→  const addRecent = useCallback((id: string) => {
   366→    setRecent(prev => [id, ...prev.filter(x => x !== id)].slice(0, 5));
   367→  }, []);
   368→
   369→  const openToolById = useCallback((id: string) => {
   370→    const t = TOOLS.find(x => x.id === id);
   371→    if (!t) return;
   372→    setOpenTool(t);
   373→    addRecent(id);
   374→  }, [addRecent]);
   375→
   376→  // Toggle a tool in/out of the comparison tray (max 2). When the tray is full
   377→  // and the tool isn't already in it, show a toast and bail.
   378→  const toggleCompare = useCallback((id: string) => {
   379→    setCompareTray(prev => {
   380→      if (prev.includes(id)) {
   381→        return prev.filter(x => x !== id);
   382→      }
   383→      if (prev.length >= 2) {
   384→        toast({
   385→          title: 'Comparison tray is full (max 2)',
   386→          description: 'Clear it first to add a different tool.',
   387→        });
   388→        return prev;
   389→      }
   390→      return [...prev, id];
   391→    });
   392→  }, []);
   393→
   394→  const clearCompareTray = useCallback(() => {
   395→    setCompareTray([]);
   396→    setCompareOpen(false);
   397→  }, []);
   398→
   399→  // Resolve the two tools currently in the tray (in tray order, not TOOLS order).
   400→  const compareTools = useMemo(
   401→    () => compareTray.map(id => TOOLS.find(t => t.id === id)).filter(Boolean) as ToolMeta[],
   402→    [compareTray],
   403→  );
   404→
   405→  const filtered = useMemo(() => {
   406→    const q = search.toLowerCase().trim();
   407→    return TOOLS.filter(t => {
   408→      if (activeCategory !== 'All' && t.category !== activeCategory) return false;
   409→      if (!q) return true;
   410→      return (t.name + ' ' + t.description + ' ' + t.benefit + ' ' + t.category).toLowerCase().includes(q);
   411→    });
   412→  }, [search, activeCategory]);
   413→
   414→  // Group filtered tools by category for the landscape grid
   415→  const grouped = useMemo(() => {
   416→    return CATEGORIES.map(c => ({
   417→      category: c,
   418→      tools: filtered.filter(t => t.category === c),
   419→    })).filter(g => g.tools.length > 0);
   420→  }, [filtered]);
   421→
   422→  // Global keyboard shortcuts — registered on window. See KeyboardShortcutsHelp
   423→  // for the user-facing list. The handler is re-bound whenever any of its
   424→  // dependencies change so the closures always see fresh state.
   425→  useEffect(() => {
   426→    const isTypingTarget = (el: Element | null) => {
   427→      if (!el) return false;
   428→      if (el instanceof HTMLInputElement) return true;
   429→      if (el instanceof HTMLTextAreaElement) return true;
   430→      if ((el as HTMLElement).isContentEditable) return true;
   431→      return false;
   432→    };
   433→
   434→    const handler = (e: KeyboardEvent) => {
   435→      const mod = e.ctrlKey || e.metaKey;
   436→
   437→      // Ctrl+K / Cmd+K — focus the search input (always, even while typing elsewhere)
   438→      if (mod && (e.key === 'k' || e.key === 'K')) {
   439→        e.preventDefault();
   440→        searchInputRef.current?.focus();
   441→        searchInputRef.current?.select();
   442→        return;
   443→      }
   444→
   445→      // Esc — if a dialog is open, let Radix handle it; otherwise clear search.
   446→      if (e.key === 'Escape') {
   447→        if (openTool || compareOpen) return; // Radix closes the dialog
   448→        if (search) {
   449→          setSearch('');
   450→          searchInputRef.current?.focus();
   451→        }
   452→        return;
   453→      }
   454→
