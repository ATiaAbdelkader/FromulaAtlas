   344→        if (minus && minus.length) {

   352→    function normalizeFormula(raw) {
   353→      const text = String(raw || '').trim();
   354→      if (!text) return '';
   355→      let out = text.split('').map(function(ch) { return SUBSCRIPT_MAP[ch] || ch; }).join('')
   356→        .replace(/[−]/g, '-').replace(/\s+/g, '')
   357→        .replace(/\{/g, '(').replace(/\}/g, ')').replace(/\[/g, '(').replace(/\]/g, ')')
   358→        .replace(/:/g, '·').replace(/\)\(/g, ')*(');
   359→      // + o * entre bloques (hidrato / sal doble). No toca cargas iónicas sueltas (NH4+ al final).
   360→      out = out.replace(/([0-9)\]])([\*+])([0-9(A-Z])/g, '$1·$3');
   361→      return out.replace(/\.+/g, '.');
   362→    }

   368→    function parseNumber(str, index) {
   369→      let i = index;
   370→      let seenDot = false;
   371→      let token = '';
   372→      while (i < str.length) {
   373→        const ch = str[i];
   374→        if (ch >= '0' && ch <= '9') { token += ch; i += 1; continue; }
   375→        if (ch === '.' && !seenDot) { seenDot = true; token += ch; i += 1; continue; }
   376→        break;
   377→      }
   378→      if (!token) return { value: 1, index: i };
   379→      const val = Number(token);
   380→      if (!Number.isFinite(val) || val <= 0) throw new Error('Número inválido en fórmula.');
   381→      return { value: val, index: i };
   382→    }

   384→    function parseTerms(str, index, stopChar) {
   385→      const terms = [];
   386→      let i = index;
   387→      while (i < str.length) {
   388→        const ch = str[i];
   389→        if (stopChar && ch === stopChar) return { terms: terms, index: i + 1 };
   390→        if (ch === '+' || ch === '-' || ch === '^') { i += 1; continue; }
   391→        if (ch === '(') {
   392→          const nested = parseTerms(str, i + 1, ')');
   393→          const cnt = parseNumber(str, nested.index);
   394→          terms.push({ type: 'group', terms: nested.terms, count: cnt.value });
   395→          i = cnt.index;
   396→          continue;
   397→        }
   398→        if (ch >= 'A' && ch <= 'Z') {
   399→          let symbol = ch;
   400→          i += 1;
   401→          if (i < str.length && str[i] >= 'a' && str[i] <= 'z') { symbol += str[i]; i += 1; }
   402→          if (!Object.prototype.hasOwnProperty.call(ATOMIC_WEIGHTS, symbol)) {
   403→            throw new Error('Elemento no reconocido: ' + symbol);
   404→          }
   405→          const cnt = parseNumber(str, i);
   406→          terms.push({ type: 'element', symbol: symbol, count: cnt.value });
   407→          i = cnt.index;
   408→          continue;
   409→        }
   410→        throw new Error('Carácter inválido en fórmula: ' + ch);
   411→      }
   412→      if (stopChar) throw new Error('Paréntesis sin cerrar en fórmula.');
   413→      return { terms: terms, index: i };
   414→    }

   416→    function collectElements(terms, mult, out) {
   417→      terms.forEach(function(t) {
   418→        if (t.type === 'element') {
   419→          out[t.symbol] = (out[t.symbol] || 0) + (t.count * mult);
   420→        } else {
   421→          collectElements(t.terms, mult * t.count, out);
   422→        }
   423→      });
   424→    }

   426→    function countMotifInTerms(terms, mult, motif) {
   427→      let total = 0;
   428→      for (let i = 0; i <= terms.length - motif.length; i += 1) {
   429→        let ok = true;
   430→        for (let j = 0; j < motif.length; j += 1) {
   431→          const term = terms[i + j];
   432→          const m = motif[j];
   433→          if (!term || term.type !== 'element' || term.symbol !== m.symbol || Math.abs(term.count - m.count) > 1e-9) {
   434→            ok = false;
   435→            break;
   436→          }
   437→        }
   438→        if (ok) total += mult;
   439→      }
   440→      terms.forEach(function(t) {
   441→        if (t.type === 'group') total += countMotifInTerms(t.terms, mult * t.count, motif);
   442→      });
   443→      return total;
   444→    }

   463→      if (/^NH4NO3$/i.test(b)) return 'Nitrato de amonio NH₄NO₃';

   468→    function parseCompound(rawFormula) {
   469→      const normalized = normalizeFormula(rawFormula);
   470→      if (!normalized) throw new Error('Escribe una fórmula.');
   471→      const parts = normalized.split(/[·•.]/).filter(Boolean);
   472→      if (!parts.length) throw new Error('Fórmula no válida.');
   473→      const segments = [];
   474→      const blocks = [];
   475→      parts.forEach(function(part) {
   476→        const m = part.match(/^(\d+(?:\.\d+)?)(.*)$/);
   477→        let coef = 1;
   478→        let body = part;
   479→        if (m && m[2]) { coef = Number(m[1]); body = m[2]; }
   480→        const parsed = parseTerms(body, 0, null);
   481→        if (parsed.index !== body.length) throw new Error('Error al parsear la fórmula.');
   482→        segments.push({ coef: coef, terms: parsed.terms, body: body, raw: part });
   483→        const unitMw = blockUnitMw(parsed.terms);
   484→        const mass = unitMw * coef;
   485→        blocks.push({
   486→          label: formatBlockLabel(part),
   487→          name: blockDisplayName(body),
   488→          coef: coef,
   489→          unitMw: unitMw,
   490→          mass: mass
   491→        });
   492→      });
   493→      const atomCounts = {};
   494→      segments.forEach(function(seg) { collectElements(seg.terms, seg.coef, atomCounts); });
   495→      let mw = 0;
   496→      Object.keys(atomCounts).forEach(function(sym) { mw += atomCounts[sym] * ATOMIC_WEIGHTS[sym]; });
   497→      if (!(mw > 0)) throw new Error('No se pudo calcular el peso molecular.');
   498→      blocks.forEach(function(blk) { blk.pct = (blk.mass / mw) * 100; });
   499→
   500→      const NO3 = [{ symbol: 'N', count: 1 }, { symbol: 'O', count: 3 }];
   501→      const NH4 = [{ symbol: 'N', count: 1 }, { symbol: 'H', count: 4 }];
   502→      let no3 = 0;
   503→      let nh4 = 0;
   504→      segments.forEach(function(seg) {
   505→        no3 += countMotifInTerms(seg.terms, seg.coef, NO3);
   506→        nh4 += countMotifInTerms(seg.terms, seg.coef, NH4);
   507→      });
   508→      const caNitrateUnits = segments.reduce(function(sum, seg) {
   509→        if (/Ca\(NO3\)2/i.test(seg.body)) return sum + seg.coef;
   510→        return sum;
   511→      }, 0);
   512→
   513→      return {
   514→        normalized: normalized,
   515→        atomCounts: atomCounts,
   516→        mw: mw,
   517→        blocks: blocks,
   518→        ions: {
   519→          no3: no3,
   520→          nh4: nh4,
   521→          ca: atomCounts.Ca || 0,
   522→          caNitrateUnits: caNitrateUnits
   523→        }
   524→      };
   525→    }

   527→    function inferMoleculeEquivalent(atomCounts, mw) {
   528→      const h = atomCounts.H || 0;
   529→      const o = atomCounts.O || 0;
   530→      const p = atomCounts.P || 0;
   531→      const s = atomCounts.S || 0;
   532→      const b = atomCounts.B || 0;
   533→      const cl = atomCounts.Cl || 0;
   534→      if (h > 0 && s > 0 && Math.abs(h / s - 2) < 0.01) {
   535→        return { n: h, label: 'Ácido sulfúrico (2 H⁺ por S)' };
   536→      }
   537→      if (h > 0 && p > 0 && h % p === 0 && h / p <= 3) {
   538→        return { n: h, label: 'Ácido fosforoso (H/P)' };
   539→      }
   540→      if (h > 0 && b > 0 && h % b === 0) {
   541→        return { n: h, label: 'Ácido bórico (H/B)' };
   542→      }
   543→      if (h > 0 && cl > 0 && h === cl) {
   544→        return { n: h, label: 'Ácido clorhídrico' };
   545→      }
   546→      if (h > 0 && o > 0 && h % 2 === 0 && h / 2 === o && !p && !s && !b) {
   547→        return { n: h, label: 'Hidróxido (OH⁻)' };
   548→      }
   549→      return null;
   550→    }
