   408→    function irrFreeMigrateLegacyIrrigationUnit() {
   409→      var NB = window.NpIrrBalance;
   410→      var appliedEl = document.getElementById('irr-applied');
   411→      var unitEl = document.getElementById('irr-unit');

   418→      var irrAreaEl = document.getElementById('irr-area');

   420→        irrAreaEl && irrAreaEl.value !== '' && Number.isFinite(parseFloat(irrAreaEl.value))
   421→          ? parseFloat(irrAreaEl.value)
   422→          : cropHa;
   423→      var legacyUnit = unitEl && unitEl.value === 'mm' ? 'mm' : 'm3';
   424→      var m3 = NB.migrateIrrigationValueToM3(parseFloat(appliedEl.value), legacyUnit, irrHa, cropHa);
   425→      if (m3 != null) appliedEl.value = String(m3);
   426→      if (unitEl) unitEl.value = 'm3';

   496→    async function irrFreeFetchClimate() {
   497→      var c = irrReadCoords();

   499→        alert('Indica latitud y longitud o elige un punto en el mapa.');

   506→        irrRolling = await window.NpIrrBalance.fetchRollingOpenMeteo(c.lat, c.lng);
   507→        irrRestoreClimateMeta();

   509→        irrFreeUpdate();
   510→        if (irrPersist) irrPersist.saveNow();
   511→      } catch (e) {
   512→        console.warn(e);
   513→        irrSetStatus('⚠️ No se pudo obtener clima. Revisa conexión o coordenadas.');
   514→        alert('No se pudo obtener el clima en ese punto. Revisa la conexión o las coordenadas.');
