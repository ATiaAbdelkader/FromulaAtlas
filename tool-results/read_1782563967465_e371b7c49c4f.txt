   188вЖТ   188вЖТ/** Convert between measure units. Returns null when conversion is blocked (e.g. soilвЖФsolution). */
   189вЖТ   189вЖТexport function convertMeasure(
   190вЖТ   190вЖТ  value: number,
   191вЖТ   191вЖТ  from: UnitDef,
   192вЖТ   192вЖТ  to: UnitDef,
   193вЖТ   193вЖТ  category: string
   194вЖТ   194вЖТ): number | null {
   195вЖТ   195вЖТ  if (category === 'temperature') {
   196вЖТ   196вЖТ    let c: number;
   197вЖТ   197вЖТ    if (from.id === 'F') c = (value - 32) * 5 / 9;
   198вЖТ   198вЖТ    else if (from.id === 'K') c = value - 273.15;
   199вЖТ   199вЖТ    else c = value;
   200вЖТ   200вЖТ    if (to.id === 'F') return c * 9 / 5 + 32;
   201вЖТ   201вЖТ    if (to.id === 'K') return c + 273.15;
   202вЖТ   202вЖТ    return c;
   203вЖТ   203вЖТ  }
   204вЖТ   204вЖТ  if (from.group && to.group && from.group !== to.group) return null;
   205вЖТ   205вЖТ  const base = value * from.toBase;
   206вЖТ   206вЖТ  return base / to.toBase;
   207вЖТ   207вЖТ}

   285вЖТ   285вЖТexport function svpFromTemp(t: number): number {
   286вЖТ   286вЖТ  return 0.6108 * Math.exp((17.27 * t) / (t + 237.3));

   289вЖТ   289вЖТexport function satAbsHum(t: number): number {
   290вЖТ   290вЖТ  return 5.018 + 0.32321 * t + 0.0081847 * t * t + 0.00031243 * t * t * t;
   291вЖТ   291вЖТ}

   303вЖТ   303вЖТ
   304вЖТ   304вЖТexport function calcVpdSimple(airTemp: number, humidity: number): VpdResult {
   305вЖТ   305вЖТ  const es = svpFromTemp(airTemp);
   306вЖТ   306вЖТ  const ea = es * (humidity / 100);
   307вЖТ   307вЖТ  const vpd = es - ea;
   308вЖТ   308вЖТ  const hd = satAbsHum(airTemp) * ((100 - humidity) / 100);
   309вЖТ   309вЖТ  return { vpd: Math.round(vpd * 100) / 100, hd: Math.round(hd * 100) / 100 };
   310вЖТ   310вЖТ}
   311вЖТ   311вЖТ
   312вЖТ   312вЖТexport function calcVpdAdvanced(
   313вЖТ   313вЖТ  airTemp: number,
   314вЖТ   314вЖТ  humidity: number,
   315вЖТ   315вЖТ  leafTemp: number
   316вЖТ   316вЖТ): VpdResult {
   317вЖТ   317вЖТ  const esLeaf = svpFromTemp(leafTemp);
   318вЖТ   318вЖТ  const esAir = svpFromTemp(airTemp);
   319вЖТ   319вЖТ  const ea = esAir * (humidity / 100);
   320вЖТ   320вЖТ  const vpd = esLeaf - ea;
   321вЖТ   321вЖТ  const hd = satAbsHum(leafTemp) - satAbsHum(airTemp) * (humidity / 100);
   322вЖТ   322вЖТ  return {
   323вЖТ   323вЖТ    vpd: Math.round(vpd * 100) / 100,
   324вЖТ   324вЖТ    hd: Math.round(hd * 100) / 100,
   325вЖТ   325вЖТ    leafTemp,
   326вЖТ   326вЖТ  };
   327вЖТ   327вЖТ}

   469вЖТ   469вЖТexport function normalizeFormula(input: string): string {
   470вЖТ   470вЖТ  let s = String(input);
   471вЖТ   471вЖТ  for (const [k, v] of Object.entries(SUBSCRIPT_MAP)) s = s.split(k).join(v);
   472вЖТ   472вЖТ  s = s.replace(/вИТ/g, '-').replace(/\s+/g, '');
   473вЖТ   473вЖТ  s = s.replace(/[\{\[]/g, '(').replace(/[\}\]]/g, ')');
   474вЖТ   474вЖТ  s = s.replace(/:/g, '¬Ј');
   475вЖТ   475вЖТ  s = s.replace(/\)\(/g, ')*(');
   476вЖТ   476вЖТ  s = s.replace(/([0-9)\]])([\*+])([0-9(A-Z])/g, '$1¬Ј$3');
   477вЖТ   477вЖТ  s = s.replace(/\.+/g, '¬Ј');

   488вЖТ   488вЖТ  while (j < s.length && /[0-9.]/.test(s[j])) { str += s[j]; j++; }
   489вЖТ   489вЖТ  return [str ? parseFloat(str) : 1, j];
   490вЖТ   490вЖТ}
   491вЖТ   491вЖТ
   492вЖТ   492вЖТfunction parseTerms(s: string, i: number, stop: string | null): [Term[], number] {
   493вЖТ   493вЖТ  const terms: Term[] = [];
   494вЖТ   494вЖТ  while (i < s.length) {
   495вЖТ   495вЖТ    const c = s[i];
   496вЖТ   496вЖТ    if (stop && c === stop) return [terms, i + 1];
   497вЖТ   497вЖТ    if (c === '(') {
   498вЖТ   498вЖТ      const [inner, j] = parseTerms(s, i + 1, ')');
   499вЖТ   499вЖТ      const [count, k] = parseNumber(s, j);
   500вЖТ   500вЖТ      terms.push({ type: 'group', terms: inner, count });
   501вЖТ   501вЖТ      i = k;
   502вЖТ   502вЖТ    } else if (/[A-Z]/.test(c)) {
   503вЖТ   503вЖТ      let sym = c;
   504вЖТ   504вЖТ      let j = i + 1;
   505вЖТ   505вЖТ      if (j < s.length && /[a-z]/.test(s[j])) { sym += s[j]; j++; }
   506вЖТ   506вЖТ      const [count, k] = parseNumber(s, j);
   507вЖТ   507вЖТ      terms.push({ type: 'element', symbol: sym, count });
   508вЖТ   508вЖТ      i = k;
   509вЖТ   509вЖТ    } else if (c === '+' || c === '-' || c === '^') {
   510вЖТ   510вЖТ      i++;
   511вЖТ   511вЖТ    } else {
   512вЖТ   512вЖТ      throw new Error(`Invalid char in formula: ${c}`);
   513вЖТ   513вЖТ    }
   514вЖТ   514вЖТ  }
   515вЖТ   515вЖТ  if (stop) throw new Error('Unclosed parenthesis');
   516вЖТ   516вЖТ  return [terms, i];
   517вЖТ   517вЖТ}
   518вЖТ   518вЖТ
   519вЖТ   519вЖТfunction collectTerms(terms: Term[], mult: number, out: Record<string, number>) {
   520вЖТ   520вЖТ  for (const t of terms) {
   521вЖТ   521вЖТ    if (t.type === 'element') {
   522вЖТ   522вЖТ      out[t.symbol] = (out[t.symbol] || 0) + t.count * mult;
   523вЖТ   523вЖТ    } else {
   524вЖТ   524вЖТ      collectTerms(t.terms, mult * t.count, out);
   525вЖТ   525вЖТ    }
   526вЖТ   526вЖТ  }
   527вЖТ   527вЖТ}
   528вЖТ   528вЖТ
   529вЖТ   529вЖТfunction countMotif(terms: Term[], mult: number, motif: { symbol: string; count: number }[]): number {
   530вЖТ   530вЖТ  let total = 0;
   531вЖТ   531вЖТ  for (let i = 0; i + motif.length <= terms.length; i++) {
   532вЖТ   532вЖТ    let ok = true;
   533вЖТ   533вЖТ    for (let j = 0; j < motif.length; j++) {
   534вЖТ   534вЖТ      const t = terms[i + j];
   535вЖТ   535вЖТ      if (t.type !== 'element' || t.symbol !== motif[j].symbol || Math.abs(t.count - motif[j].count) > 1e-9) {
   536вЖТ   536вЖТ        ok = false; break;
   537вЖТ   537вЖТ      }
   538вЖТ   538вЖТ    }
   539вЖТ   539вЖТ    if (ok) total += mult;
   540вЖТ   540вЖТ  }
   541вЖТ   541вЖТ  for (const t of terms) {
   542вЖТ   542вЖТ    if (t.type === 'group') total += countMotif(t.terms, mult * t.count, motif);
   543вЖТ   543вЖТ  }
   544вЖТ   544вЖТ  return total;
   545вЖТ   545вЖТ}
   546вЖТ   546вЖТ
   547вЖТ   547вЖТexport interface CompositionResult {
   548вЖТ   548вЖТ  ok: boolean;
   549вЖТ   549вЖТ  mw?: number;
   550вЖТ   550вЖТ  comp?: Record<string, number>;
   551вЖТ   551вЖТ  error?: string;
   552вЖТ   552вЖТ  normalized?: string;
   553вЖТ   553вЖТ}
   554вЖТ   554вЖТ
   555вЖТ   555вЖТ/** Compute elemental composition (%) from a chemical formula string. */
   556вЖТ   556вЖТexport function compFromFormula(input: string): CompositionResult {
   557вЖТ   557вЖТ  try {
   558вЖТ   558вЖТ    const normalized = normalizeFormula(input);
   559вЖТ   559вЖТ    if (!normalized) return { ok: false, error: 'Empty formula' };
   560вЖТ   560вЖТ    const segments = normalized.split(/[¬ЈвАҐ.]/).filter(Boolean);
   561вЖТ   561вЖТ    const atoms: Record<string, number> = {};
   562вЖТ   562вЖТ    let no3 = 0, nh4 = 0;
   563вЖТ   563вЖТ    let mw = 0;
   564вЖТ   564вЖТ
   565вЖТ   565вЖТ    for (const seg of segments) {
   566вЖТ   566вЖТ      let coef = 1, body = seg;
   567вЖТ   567вЖТ      const m = seg.match(/^(\d+(?:\.\d+)?)(.*)$/);
   568вЖТ   568вЖТ      if (m && m[2]) { coef = parseFloat(m[1]); body = m[2]; }
   569вЖТ   569вЖТ      const [terms] = parseTerms(body, 0, null);
   570вЖТ   570вЖТ      const segAtoms: Record<string, number> = {};
   571вЖТ   571вЖТ      collectTerms(terms, coef, segAtoms);
   572вЖТ   572вЖТ      for (const [sym, cnt] of Object.entries(segAtoms)) {
   573вЖТ   573вЖТ        atoms[sym] = (atoms[sym] || 0) + cnt;
   574вЖТ   574вЖТ        if (ATOMIC_WEIGHTS[sym]) mw += cnt * ATOMIC_WEIGHTS[sym];
   575вЖТ   575вЖТ      }
   576вЖТ   576вЖТ      no3 += countMotif(terms, coef, [{ symbol: 'N', count: 1 }, { symbol: 'O', count: 3 }]);
   577вЖТ   577вЖТ      nh4 += countMotif(terms, coef, [{ symbol: 'N', count: 1 }, { symbol: 'H', count: 4 }]);
   578вЖТ   578вЖТ    }
   579вЖТ   579вЖТ
   580вЖТ   580вЖТ    if (mw <= 0) return { ok: false, error: 'Could not compute molecular weight' };
   581вЖТ   581вЖТ
   582вЖТ   582вЖТ    const comp: Record<string, number> = {};
   583вЖТ   583вЖТ    const nAtoms = atoms['N'] || 0;
   584вЖТ   584вЖТ    for (const sym of ['N','P','K','Ca','Mg','S','Si','C','H','O','Zn','Fe','Mn','B','Cu','Mo']) {
   585вЖТ   585вЖТ      if (atoms[sym]) comp[sym] = (atoms[sym] * ATOMIC_WEIGHTS[sym] / mw) * 100;
   586вЖТ   586вЖТ    }
   587вЖТ   587вЖТ    // N partition into NO3 / NH4
   588вЖТ   588вЖТ    if (nAtoms > 0 && (no3 > 0 || nh4 > 0)) {
   589вЖТ   589вЖТ      let no3Atoms = Math.max(0, no3);
   590вЖТ   590вЖТ      let nh4Atoms = Math.max(0, nh4);
   591вЖТ   591вЖТ      const forms = no3Atoms + nh4Atoms;
   592вЖТ   592вЖТ      if (forms > nAtoms && forms > 0) {
   593вЖТ   593вЖТ        const k = nAtoms / forms;
   594вЖТ   594вЖТ        no3Atoms *= k; nh4Atoms *= k;
   595вЖТ   595вЖТ      }
   596вЖТ   596вЖТ      comp['N_NO3'] = comp['N'] * (no3Atoms / nAtoms);
   597вЖТ   597вЖТ      comp['N_NH4'] = comp['N'] * (nh4Atoms / nAtoms);
   598вЖТ   598вЖТ    }
   599вЖТ   599вЖТ    // Oxide equivalents
   600вЖТ   600вЖТ    comp['P2O5'] = (comp['P'] || 0) * OXIDE_FACTORS.P2O5;
   601вЖТ   601вЖТ    comp['K2O']  = (comp['K']  || 0) * OXIDE_FACTORS.K2O;
   602вЖТ   602вЖТ    comp['CaO']  = (comp['Ca'] || 0) * OXIDE_FACTORS.CaO;
   603вЖТ   603вЖТ    comp['MgO']  = (comp['Mg'] || 0) * OXIDE_FACTORS.MgO;
   604вЖТ   604вЖТ    comp['SiO2'] = (comp['Si'] || 0) * OXIDE_FACTORS.SiO2;
   605вЖТ   605вЖТ
   606вЖТ   606вЖТ    return { ok: true, mw, comp, normalized };
   607вЖТ   607вЖТ  } catch (e) {
   608вЖТ   608вЖТ    return { ok: false, error: e instanceof Error ? e.message : 'Parse error' };
   609вЖТ   609вЖТ  }
   610вЖТ   610вЖТ}

   827вЖТ   827вЖТexport function inferCompatLevel(aid: string, bid: string): 'C' | 'R' | 'I' {
   828вЖТ   828вЖТ  if (aid === bid) return 'C';
   829вЖТ   829вЖТ  const k = aid < bid ? `${aid}|${bid}` : `${bid}|${aid}`;
   830вЖТ   830вЖТ  if (COMPAT_OVERRIDE[k]) return COMPAT_OVERRIDE[k];
   831вЖТ   831вЖТ  const A = FERTILIZER_TRAITS[aid] || {};
   832вЖТ   832вЖТ  const B = FERTILIZER_TRAITS[bid] || {};
   833вЖТ   833вЖТ  const ca = A.ca_no3 || B.ca_no3;
   834вЖТ   834вЖТ  const has = (t: string) => !!(A[t] || B[t]);
   835вЖТ   835вЖТ  const isMap = aid === 'map' || bid === 'map';
   836вЖТ   836вЖТ  const isMkp = aid === 'mkp' || bid === 'mkp';
   837вЖТ   837вЖТ
   838вЖТ   838вЖТ  if (ca && (has('po4') || has('p_acid') || (has('npk') && (A.po4 || B.po4)))) {
   839вЖТ   839вЖТ    if ((A.po4 || B.po4 || A.npk || B.npk) && !has('hno3')) return 'I';
   840вЖТ   840вЖТ  }
   841вЖТ   841вЖТ  if (ca && (has('h3po4') || (A.strong_acid && A.h3po4) || (B.strong_acid && B.h3po4))) return 'I';
   842вЖТ   842вЖТ  if (ca && (has('h2so4') || A.h2so4 || B.h2so4)) return 'I';
   843вЖТ   843вЖТ
   844вЖТ   844вЖТ  if (ca && has('so4') && (A.so4 || B.so4)) {
   845вЖТ   845вЖТ    if ((A.nh4 && A.so4) || (B.nh4 && B.so4)) return 'I';
   846вЖТ   846вЖТ    if ([aid, bid].some(x => ['sop', 'sulfato_magnesio'].includes(x))) return 'I';
   847вЖТ   847вЖТ    if ([aid, bid].some(x => x === 'nks')) return 'I';
   848вЖТ   848вЖТ  }
   849вЖТ   849вЖТ
   850вЖТ   850вЖТ  if (has('metal_sulfate') && (has('po4') || isMap || isMkp)) return 'I';
   851вЖТ   851вЖТ  if ((A.chelate || B.chelate) && (isMap || isMkp || (has('po4') && (A.npk || B.npk)))) return 'R';
   852вЖТ   852вЖТ  if ([aid, bid].some(x => ['nks', 'nk_mg'].includes(x)) && (has('nh4') && has('so4'))) return 'R';
   853вЖТ   853вЖТ  if ([aid, bid].some(x => x === 'nks') && [aid, bid].some(x => x === 'sop')) return 'R';
   854вЖТ   854вЖТ  if (A.strong_acid && B.strong_acid && aid !== bid) return 'R';
   855вЖТ   855вЖТ
   856вЖТ   856вЖТ  return 'C';
   857вЖТ   857вЖТ}

  1401вЖТ  1401вЖТ/** Compute emissions for one fertilizer row (kg CO2e / ha). */

  1403вЖТ  1403вЖТ  const factor = CARBON_FACTORS.find(f => f.id === row.fertilizerId);
  1404вЖТ  1404вЖТ  let mfg = 0;
  1405вЖТ  1405вЖТ  if (row.mode === 'custom_total_mfg' && row.customTotalMfg != null) {
  1406вЖТ  1406вЖТ    mfg = row.customTotalMfg;
  1407вЖТ  1407вЖТ  } else if (row.mode === 'custom_per_kg' && row.customPerKg != null) {
  1408вЖТ  1408вЖТ    mfg = row.customPerKg * row.rateKgPerHa;
  1409вЖТ  1409вЖТ  } else if (factor) {
  1410вЖТ  1410вЖТ    mfg = factor.mfg * row.rateKgPerHa;
  1411вЖТ  1411вЖТ  }
  1412вЖТ  1412вЖТ  // Transport: road ~0.107 kg CO2e/t¬Јkm, sea ~0.016 kg CO2e/t¬Јkm (DEFRA 2024 averages)

  1416вЖТ  1416вЖТ                  + row.seaKm * seaFactor * row.rateKgPerHa;

  1418вЖТ  1418вЖТ  const field = factor ? fieldN2oCo2e(factor.nContent) * row.rateKgPerHa : 0;
