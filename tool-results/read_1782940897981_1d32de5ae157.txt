   310→export function FreeToolsSection() {
   311→  const [search, setSearch] = useState('');
   312→  const [activeCategory, setActiveCategory] = useState<ToolCategory | 'All'>('All');
   313→  const [openTool, setOpenTool] = useState<ToolMeta | null>(null);
   314→  const [showHints, setShowHints] = useState(false);
   315→  const [seasonPlanOpen, setSeasonPlanOpen] = useState(false);
   316→
   317→  // Comparison tray — holds up to 2 tool IDs for side-by-side comparison.
   318→  // Persists across tool-dialog opens/closes but not across page reloads (v1).
   319→  const [compareTray, setCompareTray] = useState<string[]>([]);
   320→  const [compareOpen, setCompareOpen] = useState(false);
   321→
   322→  // Ref to the search input — focused by the Ctrl+K / Cmd+K shortcut.
   323→  const searchInputRef = useRef<HTMLInputElement>(null);
   324→
   325→  // Favorites + Recently used (localStorage-persisted)
   326→  const FAV_KEY = 'nutriplant_tools_favorites_v1';
   327→  const REC_KEY = 'nutriplant_tools_recent_v1';
   328→  const [favorites, setFavorites] = useState<string[]>([]);
   329→  const [recent, setRecent] = useState<string[]>([]);
   330→
   331→  // Load persisted favorites + recent on mount; also handle #tool=<id> hash for shareable URLs
   332→  useEffect(() => {
   333→    try {
   334→      const fav = JSON.parse(localStorage.getItem(FAV_KEY) || '[]');
   335→      const rec = JSON.parse(localStorage.getItem(REC_KEY) || '[]');
   336→      if (Array.isArray(fav)) setFavorites(fav);
   337→      if (Array.isArray(rec)) setRecent(rec);
   338→    } catch { /* corrupt localStorage — ignore */ }
   339→    // Open tool from URL hash (#tool=<id>) on initial load — supports shared URLs
   340→    if (typeof window !== 'undefined') {
   341→      const m = window.location.hash.match(/#tool=([\w-]+)/);
   342→      if (m) {
   343→        const t = TOOLS.find(x => x.id === m[1]);
   344→        if (t) {
   345→          setOpenTool(t);
   346→          setRecent(prev => [t.id, ...prev.filter(id => id !== t.id)].slice(0, 5));
   347→        }
   348→      }
   349→    }
   350→    // eslint-disable-next-line react-hooks/exhaustive-deps
   351→  }, []);
   352→
   353→  // Persist favorites
   354→  useEffect(() => {
   355→    try { localStorage.setItem(FAV_KEY, JSON.stringify(favorites)); } catch { /* storage full or blocked */ }
   356→  }, [favorites]);
   357→
   358→  // Persist recent
   359→  useEffect(() => {
   360→    try { localStorage.setItem(REC_KEY, JSON.stringify(recent)); } catch { /* storage full or blocked */ }
   361→  }, [recent]);
   362→
   363→  const toggleFavorite = useCallback((id: string) => {
   364→    setFavorites(prev => prev.includes(id) ? prev.filter(x => x !== id) : [...prev, id]);
   365→  }, []);
   366→
   367→  const addRecent = useCallback((id: string) => {
   368→    setRecent(prev => [id, ...prev.filter(x => x !== id)].slice(0, 5));
   369→  }, []);
   370→
   371→  const openToolById = useCallback((id: string) => {
   372→    const t = TOOLS.find(x => x.id === id);
   373→    if (!t) return;
   374→    setOpenTool(t);
   375→    addRecent(id);
   376→  }, [addRecent]);
   377→
   378→  // Toggle a tool in/out of the comparison tray (max 2). When the tray is full
   379→  // and the tool isn't already in it, show a toast and bail.
   380→  const toggleCompare = useCallback((id: string) => {
   381→    setCompareTray(prev => {
   382→      if (prev.includes(id)) {
   383→        return prev.filter(x => x !== id);
   384→      }
   385→      if (prev.length >= 2) {
   386→        toast({
   387→          title: 'Comparison tray is full (max 2)',
   388→          description: 'Clear it first to add a different tool.',
   389→        });
   390→        return prev;
   391→      }
   392→      return [...prev, id];
   393→    });
   394→  }, []);
   395→
   396→  const clearCompareTray = useCallback(() => {
   397→    setCompareTray([]);
   398→    setCompareOpen(false);
   399→  }, []);
   400→
   401→  // Resolve the two tools currently in the tray (in tray order, not TOOLS order).
   402→  const compareTools = useMemo(
   403→    () => compareTray.map(id => TOOLS.find(t => t.id === id)).filter(Boolean) as ToolMeta[],
   404→    [compareTray],
   405→  );
   406→
   407→  const filtered = useMemo(() => {
   408→    const q = search.toLowerCase().trim();
   409→    return TOOLS.filter(t => {
   410→      if (activeCategory !== 'All' && t.category !== activeCategory) return false;
   411→      if (!q) return true;
   412→      return (t.name + ' ' + t.description + ' ' + t.benefit + ' ' + t.category).toLowerCase().includes(q);
   413→    });
   414→  }, [search, activeCategory]);
   415→
   416→  // Group filtered tools by category for the landscape grid
   417→  const grouped = useMemo(() => {
   418→    return CATEGORIES.map(c => ({
   419→      category: c,
   420→      tools: filtered.filter(t => t.category === c),
   421→    })).filter(g => g.tools.length > 0);
   422→  }, [filtered]);
   423→
   424→  // Global keyboard shortcuts — registered on window. See KeyboardShortcutsHelp
   425→  // for the user-facing list. The handler is re-bound whenever any of its
   426→  // dependencies change so the closures always see fresh state.
   427→  useEffect(() => {
   428→    const isTypingTarget = (el: Element | null) => {
   429→      if (!el) return false;
   430→      if (el instanceof HTMLInputElement) return true;
   431→      if (el instanceof HTMLTextAreaElement) return true;
   432→      if ((el as HTMLElement).isContentEditable) return true;
   433→      return false;
   434→    };
   435→
   436→    const handler = (e: KeyboardEvent) => {
   437→      const mod = e.ctrlKey || e.metaKey;
   438→
   439→      // Ctrl+K / Cmd+K — focus the search input (always, even while typing elsewhere)
   440→      if (mod && (e.key === 'k' || e.key === 'K')) {
   441→        e.preventDefault();
   442→        searchInputRef.current?.focus();
   443→        searchInputRef.current?.select();
   444→        return;
   445→      }
   446→
   447→      // Esc — if a dialog is open, let Radix handle it; otherwise clear search.
   448→      if (e.key === 'Escape') {
   449→        if (openTool || compareOpen) return; // Radix closes the dialog
   450→        if (search) {
   451→          setSearch('');
   452→          searchInputRef.current?.focus();
   453→        }
