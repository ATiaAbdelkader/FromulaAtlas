   297→    function calculateCIC() {
   298→      const ids = ['k-initial','ca-initial','mg-initial','h-initial','na-initial','al-initial'];
   299→      const vals = ids.map(id => parseFloat(document.getElementById(id).value || 0));
   300→      const cic = vals.reduce((a,b) => a + b, 0);
   301→      document.getElementById('cic-total').value = cic.toFixed(2);
   302→      calculateCationPercentages(cic);
   303→      updateCationStatusIcons();
   304→      calculateAutomaticAdjustments();
   305→    }

   307→    function calculateCationPercentages(cic) {
   308→      const map = [['k-initial','k-percent'],['ca-initial','ca-percent'],['mg-initial','mg-percent'],['h-initial','h-percent'],['na-initial','na-percent'],['al-initial','al-percent']];
   309→      map.forEach(([id,pid]) => {
   310→        const v = parseFloat(document.getElementById(id).value || 0);
   311→        document.getElementById(pid).textContent = cic > 0 ? ((v / cic) * 100).toFixed(1) + '%' : '0.0%';
   312→      });
   313→    }

   315→    function updateStatusIcon(elementId, percent, minRange, maxRange) {
   316→      const el = document.getElementById(elementId);
   317→      if (!el) return;
   318→      if (isNaN(percent) || percent === 0) { el.textContent = '-'; return; }
   319→      el.textContent = (percent >= minRange && percent <= maxRange) ? '✅' : (percent > maxRange ? '⚠️' : '❌');
   320→    }

   322→    function updateCationStatusIcons() {
   323→      const k = parseFloat(document.getElementById('k-initial').value || 0);
   324→      const ca = parseFloat(document.getElementById('ca-initial').value || 0);
   325→      const mg = parseFloat(document.getElementById('mg-initial').value || 0);
   326→      const h = parseFloat(document.getElementById('h-initial').value || 0);

   329→      const cic = k + ca + mg + h + na + al;

   333→      updateStatusIcon('mg-status', (mg / cic) * 100, 10, 15);
   334→      updateStatusIcon('h-status', (h / cic) * 100, 0, 10);
   335→      updateStatusIcon('na-status', (na / cic) * 100, 0, 1);
   336→      updateStatusIcon('al-status', (al / cic) * 100, 0, 1);
   337→    }

   352→    function calculateAutomaticAdjustments() {
   353→      const cic = parseFloat(document.getElementById('cic-total').value || 0);

   358→        ca: parseFloat(document.getElementById('ca-initial').value || 0),
   359→        mg: parseFloat(document.getElementById('mg-initial').value || 0),
   360→        h: parseFloat(document.getElementById('h-initial').value || 0),
   361→        na: parseFloat(document.getElementById('na-initial').value || 0),
   362→        al: parseFloat(document.getElementById('al-initial').value || 0)
   363→      };
   364→      applyAdjustmentsToFields({
   365→        k: ideal.k - current.k,
   366→        ca: ideal.ca - current.ca,
   367→        mg: ideal.mg - current.mg,
   368→        h: ideal.h - current.h,
   369→        na: ideal.na - current.na,
   370→        al: ideal.al - current.al
   371→      });
   372→    }

   507→    function convertMeqToKgHa(meq, pesoEquivalente) {
   508→      const densidadAparente = parseFloat(document.getElementById('soil-density').value || 1.1);
   509→      const profundidad = parseFloat(document.getElementById('soil-depth').value || 30) / 100;
   510→      return meq * pesoEquivalente * 10 * (100 * 100 * profundidad * densidadAparente) / 1000;
   511→    }

   513→    function convertKgHaToMeq(kgHa, pesoMolecular) {
   514→      const densidadAparente = parseFloat(document.getElementById('soil-density').value || 1.1);
   515→      const profundidad = parseFloat(document.getElementById('soil-depth').value || 30) / 100;
   516→      return kgHa / (pesoMolecular * 10 * (100 * 100 * profundidad * densidadAparente) / 1000);
   517→    }

   519→    function getSoilReachPercent() {
   520→      const el = document.getElementById('soil-reach-percent');
   521→      if (!el) return 100;
   522→      const raw = String(el.value).trim();
   523→      if (raw === '') {
   524→        el.value = '100';
   525→        return 100;
   526→      }
   527→      let v = parseFloat(raw);
   528→      if (Number.isNaN(v)) {
   529→        el.value = '100';
   530→        return 100;
   531→      }
   532→      v = Math.max(10, Math.min(100, v));
   533→      el.value = String(v);
   534→      return v;
   535→    }

   537→    function calcularEstrategiaEnmiendas(enmiendasSeleccionadas, necesidades) {
   538→      const estrategia = [];
   539→      let caRestante = necesidades.ca;
   540→      let mgRestante = necesidades.mg;
   541→      let kRestante = necesidades.k;
   542→      let so4Necesario = necesidades.so4;
   543→
   544→      const tieneDolomita = enmiendasSeleccionadas.find(a => a.id === 'dolomite');
   545→      const tieneYeso = enmiendasSeleccionadas.find(a => a.id === 'gypsum');
   546→      const tieneCalAgricola = enmiendasSeleccionadas.find(a => a.id === 'lime');
   547→      const tieneMgSO4 = enmiendasSeleccionadas.find(a => a.id === 'mgso4-mono');
   548→      const tieneSOP = enmiendasSeleccionadas.find(a => a.id === 'sop-granular');
   549→
   550→      if (tieneDolomita && caRestante > 0 && mgRestante > 0) {
   551→        const caKgHaNeeded = convertMeqToKgHa(caRestante, 20.04);
   552→        const mgKgHaNeeded = convertMeqToKgHa(mgRestante, 12.15);
   553→        const caAmount = caKgHaNeeded / ((tieneDolomita.ca || 21.7) / 100);
   554→        const mgAmount = mgKgHaNeeded / ((tieneDolomita.mg || 13.2) / 100);
   555→        const dosisDolomita = Math.max(caAmount, mgAmount);
   556→        estrategia.push({ tipo: 'dolomite', dosis: dosisDolomita });
   557→        caRestante = Math.max(0, caRestante - convertKgHaToMeq(dosisDolomita * ((tieneDolomita.ca || 21.7) / 100), 20.04));
   558→        mgRestante = Math.max(0, mgRestante - convertKgHaToMeq(dosisDolomita * ((tieneDolomita.mg || 13.2) / 100), 12.15));
   559→      }
   560→
   561→      if (caRestante > 0) {
   562→        if (tieneYeso) {
   563→          const caKgHaNeeded = convertMeqToKgHa(caRestante, 20.04);
   564→          estrategia.push({ tipo: 'gypsum', dosis: caKgHaNeeded / ((tieneYeso.ca || 23.3) / 100) });
   565→          caRestante = 0;
   566→        } else if (tieneCalAgricola) {
   567→          const caKgHaNeeded = convertMeqToKgHa(caRestante, 20.04);
   568→          estrategia.push({ tipo: 'lime', dosis: caKgHaNeeded / ((tieneCalAgricola.ca || 40) / 100) });
   569→          caRestante = 0;
   570→        }
   571→      }
   572→
   573→      if (mgRestante > 0 && tieneMgSO4) {
   574→        const mgKgHaNeeded = convertMeqToKgHa(mgRestante, 12.15);
   575→        estrategia.push({ tipo: 'mgso4-mono', dosis: mgKgHaNeeded / ((tieneMgSO4.mg || 17) / 100) });
   576→        mgRestante = 0;
   577→      }
   578→
   579→      if (kRestante > 0 && tieneSOP) {
   580→        const kKgHaNeeded = convertMeqToKgHa(kRestante, 39.1);
   581→        estrategia.push({ tipo: 'sop-granular', dosis: kKgHaNeeded / ((tieneSOP.k || 41.5) / 100) });
   582→        kRestante = 0;
   583→      }
   584→
   585→      const so4YaAportado = estrategia.reduce((t, e) => {
   586→        const am = enmiendasSeleccionadas.find(a => a.id === e.tipo);
   587→        return t + (e.dosis * ((am?.so4 || 0) / 100));
   588→      }, 0);
   589→      const so4Restante = Math.max(0, so4Necesario - so4YaAportado);
   590→      if (so4Restante > 0 && tieneYeso) estrategia.push({ tipo: 'gypsum', dosis: so4Restante / ((tieneYeso.so4 || 55.8) / 100) });
   591→
   592→      return estrategia.filter(e => e.dosis >= 100);
   593→    }

   595→    function showCombinedAmendmentResults(details, reachPercent) {
   596→      const resultsSection = document.getElementById('amendment-results');
   597→      const factor = reachPercent / 100;
   598→      const adjusted = details.map(x => ({
   599→        ...x,
   600→        amount: x.amount * factor,
   601→        k: x.k * factor,
   602→        ca: x.ca * factor,
   603→        mg: x.mg * factor,
   604→        so4: x.so4 * factor,
   605→        si: x.si * factor
   606→      }));
   607→
   608→      if (!adjusted.length) {
   609→        resultsSection.innerHTML = `<h3>📊 ${t('Resultados del Cálculo de Enmiendas', 'Amendment Calculation Results')}</h3><p>${t('No se requieren enmiendas con las seleccionadas actuales.', 'No amendments are required with the currently selected materials.')}</p>`;
   610→        resultsSection.style.display = 'block';
   611→        return;
   612→      }
   613→
   614→      const totals = adjusted.reduce((acc, a) => {
   615→        acc.k += a.k || 0; acc.ca += a.ca || 0; acc.mg += a.mg || 0; acc.so4 += a.so4 || 0; acc.si += a.si || 0;
   616→        return acc;
   617→      }, { k:0, ca:0, mg:0, so4:0, si:0 });
   618→
   619→      let html = `<h3>📊 ${t('Resultados del Cálculo de Enmiendas', 'Amendment Calculation Results')}</h3>`;
   620→      html += `<div class="amendment-summary"><h4>🎯 ${t('Aportes Totales:', 'Total Contributions:')}</h4><ul>`;
   621→      if (totals.k > 0) html += `<li><strong>${t('Potasio', 'Potassium')} (K⁺):</strong> ${formatNumber(totals.k)} kg/ha</li>`;
   622→      if (totals.ca > 0) html += `<li><strong>${t('Calcio', 'Calcium')} (Ca²⁺):</strong> ${formatNumber(totals.ca)} kg/ha</li>`;
   623→      if (totals.mg > 0) html += `<li><strong>${t('Magnesio', 'Magnesium')} (Mg²⁺):</strong> ${formatNumber(totals.mg)} kg/ha</li>`;
   624→      if (totals.so4 > 0) html += `<li><strong>${t('Sulfato', 'Sulfate')} (SO₄²⁻):</strong> ${formatNumber(totals.so4)} kg/ha</li>`;
   625→      if (totals.si > 0) html += `<li><strong>${t('Silicio', 'Silicon')} (<span class="notranslate" translate="no">Si</span>):</strong> ${formatNumber(totals.si)} kg/ha</li>`;
   626→      html += '</ul>';
   627→      html += `<div style="font-size:12px;color:#64748b;margin-top:6px;">${t('% del volumen de suelo explorado por raíces', '% of soil volume explored by roots')} (${formatNumber(reachPercent,0)}%).</div>`;
   628→      html += '</div>';
   629→      html += `<div class="amendment-details"><h4>📋 ${t('Detalles por Enmienda:', 'Amendment Details:')}</h4><table class="results-table"><thead><tr><th>${t('Enmienda', 'Amendment')}</th><th>${t('Cantidad', 'Quantity')} (kg/ha)</th><th>K⁺</th><th>Ca²⁺</th><th>Mg²⁺</th><th>SO₄²⁻</th><th><span class="notranslate" translate="no">Si</span></th></tr></thead><tbody>`;
   630→      adjusted.forEach(a => {
   631→        html += `<tr><td>${a.name}</td><td>${formatNumber(a.amount)}</td><td>${a.k > 0 ? formatNumber(a.k) : '-'}</td><td>${a.ca > 0 ? formatNumber(a.ca) : '-'}</td><td>${a.mg > 0 ? formatNumber(a.mg) : '-'}</td><td>${a.so4 > 0 ? formatNumber(a.so4) : '-'}</td><td>${a.si > 0 ? formatNumber(a.si) : '-'}</td></tr>`;
   632→      });
   633→      html += '</tbody></table></div>';
   634→      resultsSection.innerHTML = html;
   635→      resultsSection.style.display = 'block';
   636→    }

   638→    function calculateAmendment() {
   639→      const selected = getSelectedAmendments();
   640→      if (!selected.length) {
   641→        alert(t('Por favor selecciona al menos una enmienda antes de calcular', 'Please select at least one amendment before calculating'));
   642→        return;
   643→      }
   644→      const kTarget = parseFloat(document.getElementById('k-target').value || 0);
   645→      const caTarget = parseFloat(document.getElementById('ca-target').value || 0);
   646→      const mgTarget = parseFloat(document.getElementById('mg-target').value || 0);
   647→      const naTarget = parseFloat(document.getElementById('na-target').value || 0);
   648→      const soilPH = parseFloat(document.getElementById('soil-ph').value || 7);
   649→      const reachPercent = getSoilReachPercent();
   650→
   651→      const strategy = calcularEstrategiaEnmiendas(selected, {
   652→        ca: Math.max(0, caTarget),
   653→        mg: Math.max(0, mgTarget),
   654→        k: Math.max(0, kTarget),
   655→        so4: naTarget < 0 ? Math.abs(naTarget) : 0,
   656→        pH: soilPH
   657→      });
   658→
   659→      const details = strategy.map(s => {
   660→        const a = selected.find(x => x.id === s.tipo);
   661→        const amount = s.dosis;
   662→        const ca = amount * ((a?.ca || 0) / 100);
   663→        const mg = amount * ((a?.mg || 0) / 100);
   664→        const k = amount * ((a?.k || 0) / 100);
   665→        const so4 = amount * ((a?.so4 || 0) / 100);
   666→        const si = amount * ((a?.si || 0) / 100);
   667→        return { name: a?.name || s.tipo, amount, ca, mg, k, so4, si };
   668→      });
   669→
   670→      showCombinedAmendmentResults(details, reachPercent);
   671→    }
