   332→    function calculateSaturationAbsoluteHumidity(tempC) {
   333→      return 5.018 + (0.32321 * tempC) + (0.0081847 * tempC * tempC) + (0.00031243 * tempC * tempC * tempC);
   334→    }

   336→    function calculateVPDSimple(airTemp, humidity) {
   337→      const es = 0.6108 * Math.exp((17.27 * airTemp) / (airTemp + 237.3));
   338→      const ea = es * (humidity / 100);
   339→      const vpd = es - ea;
   340→      const hd = calculateSaturationAbsoluteHumidity(airTemp) * ((100 - humidity) / 100);
   341→      return { vpd: parseFloat(vpd.toFixed(2)), hd: parseFloat(hd.toFixed(2)) };
   342→    }

   344→    function calculateLeafTempFromRadiation(airTemp, solarRadiation) {
   345→      const leafTemp = solarRadiation > 200
   346→        ? airTemp + (((solarRadiation - 200) * 0.6) / 100)
   347→        : airTemp;
   348→      return parseFloat(leafTemp.toFixed(1));
   349→    }

   351→    function calculateVPDAdvanced(airTemp, airHumidity, leafTemp) {
   352→      const esLeaf = 0.6108 * Math.exp((17.27 * leafTemp) / (leafTemp + 237.3));
   353→      const esAir = 0.6108 * Math.exp((17.27 * airTemp) / (airTemp + 237.3));
   354→      const ea = esAir * (airHumidity / 100);
   355→      const vpd = esLeaf - ea;
   356→      const hd = calculateSaturationAbsoluteHumidity(leafTemp) -
   357→        (calculateSaturationAbsoluteHumidity(airTemp) * (airHumidity / 100));
   358→      return { vpd: parseFloat(vpd.toFixed(2)), hd: parseFloat(hd.toFixed(2)) };
   359→    }

   361→    function getHumidityDeficitClass(hdValue) {
   362→      const hd = Number(hdValue);
   363→      if (!Number.isFinite(hd)) return { label: t('N/D', 'N/A'), message: t('Sin dato suficiente', 'Not enough data'), color: '#64748b', bg: '#f8fafc' };
   364→      if (hd < 3) return { label: t('Muy bajo', 'Very low'), message: t('Ambiente húmedo / baja demanda evaporativa', 'Humid environment / low evaporative demand'), color: '#2563eb', bg: '#eff6ff' };
   365→      if (hd < 6) return { label: t('Bajo-moderado', 'Low-moderate'), message: t('Demanda manejable', 'Manageable demand'), color: '#16a34a', bg: '#f0fdf4' };
   366→      if (hd < 10) return { label: t('Moderado', 'Moderate'), message: t('Transpiración activa; vigilar cultivo sensible', 'Active transpiration; monitor sensitive crops'), color: '#ca8a04', bg: '#fefce8' };
   367→      if (hd <= 15) return { label: t('Alto', 'High'), message: t('Alta demanda evaporativa', 'High evaporative demand'), color: '#ea580c', bg: '#fff7ed' };
   368→      return { label: t('Muy alto', 'Very high'), message: t('Demanda muy alta; validar en campo', 'Very high demand; validate in crop'), color: '#dc2626', bg: '#fef2f2' };
   369→    }

   471→    async function vpdFreeGetWeatherData(lat, lng) {
   472→      var url = 'https://api.open-meteo.com/v1/forecast?latitude=' + encodeURIComponent(lat) +
   473→        '&longitude=' + encodeURIComponent(lng) +
   474→        '&current=temperature_2m,relative_humidity_2m,shortwave_radiation,uv_index';
   475→      var response = await fetch(url);
   476→      if (!response.ok) throw new Error('HTTP ' + response.status);
   477→      var data = await response.json();
   478→      if (!data || !data.current) throw new Error('Respuesta inválida');
   479→      var cur = data.current;
   480→      var rad = cur.shortwave_radiation;
   481→      var radNum = (rad != null && Number.isFinite(Number(rad))) ? Number(rad) : null;
   482→      var uv = cur.uv_index;
   483→      var uvNum = (uv != null && Number.isFinite(Number(uv))) ? Number(uv) : null;
   484→      return {
   485→        temperature: cur.temperature_2m,
   486→        humidity: cur.relative_humidity_2m,
   487→        shortwaveRadiation: radNum,
   488→        uvIndex: uvNum
   489→      };
   490→    }

   617→    function createVPDRangeChart(vpdValue) {

   624→      let status = t('Óptimo', 'Optimal');
   625→      if (vpdValue < optimalMin) { barColor = '#3b82f6'; status = t('Bajo', 'Low'); }
   626→      else if (vpdValue > optimalMax) { barColor = '#ef4444'; status = t('Alto', 'High'); }
