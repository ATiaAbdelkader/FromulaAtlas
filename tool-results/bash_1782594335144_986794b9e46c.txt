   304→export function FreeToolsSection() {
   305→  const [search, setSearch] = useState('');
   306→  const [activeCategory, setActiveCategory] = useState<ToolCategory | 'All'>('All');
   307→  const [openTool, setOpenTool] = useState<ToolMeta | null>(null);
   308→  const [showHints, setShowHints] = useState(false);
   309→
   310→  // Favorites + Recently used (localStorage-persisted)
   311→  const FAV_KEY = 'nutriplant_tools_favorites_v1';
   312→  const REC_KEY = 'nutriplant_tools_recent_v1';
   313→  const [favorites, setFavorites] = useState<string[]>([]);
   314→  const [recent, setRecent] = useState<string[]>([]);
   315→
   316→  // Load persisted favorites + recent on mount; also handle #tool=<id> hash for shareable URLs
   317→  useEffect(() => {
   318→    try {
   319→      const fav = JSON.parse(localStorage.getItem(FAV_KEY) || '[]');
   320→      const rec = JSON.parse(localStorage.getItem(REC_KEY) || '[]');
   321→      if (Array.isArray(fav)) setFavorites(fav);
   322→      if (Array.isArray(rec)) setRecent(rec);
   323→    } catch { /* corrupt localStorage — ignore */ }
   324→    // Open tool from URL hash (#tool=<id>) on initial load — supports shared URLs
   325→    if (typeof window !== 'undefined') {
   326→      const m = window.location.hash.match(/#tool=([\w-]+)/);
   327→      if (m) {
   328→        const t = TOOLS.find(x => x.id === m[1]);
   329→        if (t) {
   330→          setOpenTool(t);
   331→          setRecent(prev => [t.id, ...prev.filter(id => id !== t.id)].slice(0, 5));
   332→        }
   333→      }
   334→    }
   335→    // eslint-disable-next-line react-hooks/exhaustive-deps
   336→  }, []);
   337→
   338→  // Persist favorites
   339→  useEffect(() => {
   340→    try { localStorage.setItem(FAV_KEY, JSON.stringify(favorites)); } catch { /* storage full or blocked */ }
   341→  }, [favorites]);
   342→
   343→  // Persist recent
   344→  useEffect(() => {
   345→    try { localStorage.setItem(REC_KEY, JSON.stringify(recent)); } catch { /* storage full or blocked */ }
   346→  }, [recent]);
   347→
   348→  const toggleFavorite = useCallback((id: string) => {
   349→    setFavorites(prev => prev.includes(id) ? prev.filter(x => x !== id) : [...prev, id]);
   350→  }, []);
   351→
   352→  const addRecent = useCallback((id: string) => {
   353→    setRecent(prev => [id, ...prev.filter(x => x !== id)].slice(0, 5));
   354→  }, []);
   355→
   356→  const openToolById = useCallback((id: string) => {
   357→    const t = TOOLS.find(x => x.id === id);
   358→    if (!t) return;
   359→    setOpenTool(t);
   360→    addRecent(id);
   361→  }, [addRecent]);
   362→
   363→  const filtered = useMemo(() => {
   364→    const q = search.toLowerCase().trim();
   365→    return TOOLS.filter(t => {
   366→      if (activeCategory !== 'All' && t.category !== activeCategory) return false;
   367→      if (!q) return true;
   368→      return (t.name + ' ' + t.description + ' ' + t.benefit + ' ' + t.category).toLowerCase().includes(q);
   369→    });
   370→  }, [search, activeCategory]);
   371→
   372→  // Group filtered tools by category for the landscape grid
   373→  const grouped = useMemo(() => {
   374→    return CATEGORIES.map(c => ({
   375→      category: c,
   376→      tools: filtered.filter(t => t.category === c),
   377→    })).filter(g => g.tools.length > 0);
   378→  }, [filtered]);
   379→
   380→  return (
   381→    <section className="space-y-5">
   382→      {/* Hero header */}
   383→      <div className="rounded-xl p-5 sm:p-6 bg-gradient-to-br from-emerald-700 via-green-700 to-teal-800 text-white">
   384→        <div className="flex items-center gap-2 mb-2 text-emerald-100 text-xs font-medium uppercase tracking-wide">
   385→          <Sparkles className="h-3.5 w-3.5" /> 18 Free Agronomic Tools
   386→        </div>
   387→        <h2 className="text-xl sm:text-2xl font-bold leading-tight mb-1">NutriPlant PRO Free Tools</h2>
   388→        <p className="text-sm text-emerald-100/90 max-w-2xl">
   389→          A native reimplementation of NutriPlant PRO's public free-tools collection —
   390→          converters, solution & water diagnostics, fertilizer calculators, soil & irrigation
   391→          tools, and quick-reference matrices. All calculations run client-side; nothing is sent to a server.
   392→        </p>
   393→        <div className="flex flex-wrap gap-3 mt-3 text-xs">
   394→          {CATEGORIES.map(c => {
   395→            const count = TOOLS.filter(t => t.category === c).length;
   396→            return (
   397→              <div key={c} className="bg-white/10 backdrop-blur rounded px-2 py-1 border border-white/20 flex items-center gap-1.5">
   398→                <span className="w-1.5 h-1.5 rounded-full" style={{ background: CATEGORY_DOT_COLORS[c] }} />
   399→                <span className="text-emerald-100">{c}</span> <span className="font-bold ml-0.5">{count}</span>
   400→              </div>
   401→            );
   402→          })}
   403→        </div>
   404→      </div>
   405→
   406→      {/* Introduction panel — What / How / Why */}
   407→      <div className="grid grid-cols-1 md:grid-cols-3 gap-3">
   408→        <IntroCard
   409→          icon={Info}
   410→          color="#0891b2"
   411→          title="What these tools are"
   412→          body="A curated set of 18 calculators and reference tables covering the day-to-day workflow of agronomists, growers, and consultants: unit conversions, water & nutrient solution diagnostics, fertilizer formulation, soil & irrigation planning, and crop-nutrition reference data."
   413→        />
   414→        <IntroCard
   415→          icon={BookOpen}
   416→          color="#16a34a"
   417→          title="How to use them"
   418→          body="Browse the cards below, filter by category, or search by name. Click any card to open the tool in a dialog — enter your inputs and the result updates live. Each tool's dialog also shows a short benefit and step-by-step usage hint."
   419→        />
   420→        <IntroCard
   421→          icon={Lightbulb}
   422→          color="#d97706"
   423→          title="Why use them"
   424→          body="They replace guesswork with calculation: precise acid doses, exact NPK blends, FAO-56 irrigation sheets, CEC-based amendment plans, and fertilization carbon footprints — turning lab reports and field observations into defensible decisions."
   425→        />
   426→      </div>
   427→
   428→      {/* Search & filter */}
   429→      <div className="sticky top-[120px] z-20 bg-background/95 backdrop-blur rounded-lg border border-border p-3">
   430→        <div className="flex flex-wrap items-center gap-2">
   431→          <div className="relative flex-1 min-w-[200px]">
   432→            <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground pointer-events-none" />
   433→            <Input
   434→              value={search}
   435→              onChange={e => setSearch(e.target.value)}
   436→              placeholder="Search tools by name, description, or benefit..."
   437→              className="pl-9 pr-8 h-10"
   438→            />
   439→            {search && (
   440→              <button onClick={() => setSearch('')} className="absolute right-3 top-1/2 -translate-y-1/2 text-muted-foreground hover:text-foreground">
   441→                <X className="h-4 w-4" />
   442→              </button>
   443→            )}
   444→          </div>
   445→          <div className="flex flex-wrap gap-1">
   446→            <CategoryChip active={activeCategory === 'All'} onClick={() => setActiveCategory('All')} label="All" count={TOOLS.length} />
   447→            {CATEGORIES.map(c => (
   448→              <CategoryChip
   449→                key={c}
   450→                active={activeCategory === c}
   451→                onClick={() => setActiveCategory(c)}
   452→                label={c}
   453→                count={TOOLS.filter(t => t.category === c).length}
   454→              />
   455→            ))}
   456→          </div>
   457→        </div>
   458→      </div>
   459→
   460→      {/* Favorites + Recently used (auto-hide when empty) */}
   461→      {(favorites.length > 0 || recent.length > 0) && (
   462→        <div className="space-y-3">
   463→          {favorites.length > 0 && (
   464→            <ToolRow
   465→              label="Favorites"
   466→              icon={<Star className="h-3.5 w-3.5 fill-amber-400 text-amber-400" />}
   467→              ids={favorites}
   468→              onOpen={openToolById}
   469→            />
   470→          )}
   471→          {recent.length > 0 && (
   472→            <ToolRow
   473→              label="Recently used"
   474→              icon={<Clock className="h-3.5 w-3.5 text-emerald-600" />}
   475→              ids={recent}
   476→              onOpen={openToolById}
   477→            />
   478→          )}
   479→        </div>
   480→      )}
   481→
   482→      {/* Tools grouped by category — landscape cards */}
   483→      <div className="space-y-6">
   484→        {grouped.map(group => (
   485→          <div key={group.category} className="space-y-3">
   486→            <div className="flex items-center gap-2">
   487→              <div className="w-2 h-6 rounded-sm" style={{ background: CATEGORY_DOT_COLORS[group.category] }} />
   488→              <h3 className="text-base font-bold tracking-tight">{group.category}</h3>
   489→              <Badge variant="secondary" className="text-[10px] font-mono">{group.tools.length}</Badge>
   490→            </div>
   491→            <div className="grid grid-cols-1 lg:grid-cols-2 gap-3">
   492→              {group.tools.map(tool => (
   493→                <LandscapeToolCard
   494→                  key={tool.id}
   495→                  tool={tool}
   496→                  onOpen={() => { setOpenTool(tool); addRecent(tool.id); }}
   497→                  isFavorite={favorites.includes(tool.id)}
   498→                  onToggleFavorite={() => toggleFavorite(tool.id)}
   499→                />
   500→              ))}
   501→            </div>
   502→          </div>
   503→        ))}
   504→      </div>
   505→
   506→      {filtered.length === 0 && (
   507→        <div className="text-center py-12">
   508→          <div className="rounded-full bg-muted p-4 inline-flex mb-3">
   509→            <Search className="h-8 w-8 text-muted-foreground" />
   510→          </div>
   511→          <h3 className="text-base font-semibold mb-1">No tools match</h3>
   512→          <p className="text-sm text-muted-foreground">Try a different search or category.</p>
   513→        </div>
   514→      )}
   515→
   516→      {/* Tool dialog — near-fullscreen like NutriPlant PRO: header + full-height tool area */}
   517→      <Dialog open={!!openTool} onOpenChange={open => !open && setOpenTool(null)}>
   518→        <DialogContent className="!max-w-[1600px] w-[98vw] !max-h-[96vh] h-[96vh] overflow-hidden p-0 gap-0 flex flex-col">
   519→          {/* Compact header bar — title + collapsible hints toggle */}
   520→          <DialogHeader className="px-5 py-3 border-b border-border flex-shrink-0">
   521→            <div className="flex items-center justify-between gap-3">
   522→              <DialogTitle className="flex items-center gap-2 text-base">
   523→                {openTool && (
   524→                  <span
   525→                    className="flex items-center justify-center h-8 w-8 rounded-lg"
   526→                    style={{ background: `${CATEGORY_DOT_COLORS[openTool.category]}20`, color: CATEGORY_DOT_COLORS[openTool.category] }}
   527→                  >
   528→                    <openTool.icon className="h-4 w-4" />
   529→                  </span>
   530→                )}
   531→                {openTool?.name}
   532→                {openTool && (
   533→                  <Badge variant="outline" className={`text-[10px] ml-1 ${CATEGORY_COLORS[openTool.category]}`}>
   534→                    {openTool.category}
   535→                  </Badge>
   536→                )}
   537→              </DialogTitle>
   538→              <div className="flex items-center gap-2">
   539→                <Button
   540→                  size="sm"
   541→                  variant="ghost"
   542→                  className="h-8 gap-1.5 text-xs"
   543→                  onClick={() => setShowHints(v => !v)}
   544→                >
   545→                  <HelpCircle className="h-3.5 w-3.5" />
   546→                  {showHints ? 'Hide guide' : 'Show guide'}
   547→                </Button>
   548→              </div>
   549→            </div>
   550→            <DialogDescription className="text-xs mt-1">{openTool?.description}</DialogDescription>
   551→
   552→            {/* Collapsible benefit + how-to bar — inline, doesn't steal horizontal space from the tool */}
   553→            {openTool && showHints && (
   554→              <div className="grid grid-cols-1 md:grid-cols-2 gap-2 mt-2">
   555→                <div className="rounded-md px-3 py-2 border border-emerald-200 dark:border-emerald-900 bg-emerald-50/60 dark:bg-emerald-950/20">
   556→                  <div className="flex items-center gap-1.5 text-[10px] uppercase tracking-wide text-emerald-700 dark:text-emerald-300 font-semibold mb-0.5">
   557→                    <Lightbulb className="h-3 w-3" /> Benefit
   558→                  </div>
   559→                  <p className="text-xs leading-snug text-foreground">{openTool.benefit}</p>
   560→                </div>
   561→                <div className="rounded-md px-3 py-2 border border-blue-200 dark:border-blue-900 bg-blue-50/60 dark:bg-blue-950/20">
   562→                  <div className="flex items-center gap-1.5 text-[10px] uppercase tracking-wide text-blue-700 dark:text-blue-300 font-semibold mb-0.5">
   563→                    <HelpCircle className="h-3 w-3" /> How to use
   564→                  </div>
   565→                  <ol className="text-xs leading-snug text-foreground list-decimal pl-4 space-y-0.5">
   566→                    {openTool.howToUse.map((step, i) => <li key={i}>{step}</li>)}
   567→                  </ol>
   568→                </div>
   569→              </div>
   570→            )}
   571→          </DialogHeader>
   572→
   573→          {/* Full-height tool area — gets 100% of remaining viewport height */}
   574→          <div className="flex-1 min-h-0 overflow-y-auto p-5">
   575→            {openTool && <ToolExportBar tool={openTool} />}
   576→            <div className={openTool ? 'mt-3' : ''}>
   577→              {openTool && <openTool.Component />}
   578→            </div>
   579→          </div>
   580→        </DialogContent>
   581→      </Dialog>
   582→    </section>
   583→  );
   584→}

   586→/** Landscape (horizontal) tool card — icon on the left, content on the right, action arrow on the far right. */
   587→function LandscapeToolCard({
   588→  tool, onOpen, isFavorite, onToggleFavorite,
   589→}: {
   590→  tool: ToolMeta;
   591→  onOpen: () => void;
   592→  isFavorite: boolean;
   593→  onToggleFavorite: () => void;

   595→  return (
   596→    <div
   597→      role="button"
   598→      tabIndex={0}

   678→/** Horizontal chip row for Favorites / Recently used — auto-hides when no tools match. */
   679→function ToolRow({
   680→  label, icon, ids, onOpen,
   681→}: {
   682→  label: string;
   683→  icon: React.ReactNode;
   684→  ids: string[];
   685→  onOpen: (id: string) => void;
   686→}) {
   687→  const tools = ids.map(id => TOOLS.find(t => t.id === id)).filter(Boolean) as ToolMeta[];

   689→  return (
   690→    <div className="rounded-lg border border-border bg-card p-3">
   691→      <div className="flex items-center gap-1.5 mb-2 text-[10px] font-semibold uppercase tracking-wide text-muted-foreground">
   692→        {icon}
   693→        <span>{label}</span>
   694→        <span className="font-mono opacity-70">{tools.length}</span>
   695→      </div>
   696→      <div className="flex flex-wrap gap-1.5">
   697→        {tools.map(t => {
   698→          const Icon = t.icon;
   699→          return (
   700→            <button
   701→              key={t.id}
   702→              onClick={() => onOpen(t.id)}
   703→              className="inline-flex items-center gap-1.5 rounded-full border border-border bg-background px-2.5 py-1 text-xs hover:border-emerald-400 hover:bg-emerald-50 dark:hover:bg-emerald-950/40 hover:text-emerald-700 dark:hover:text-emerald-300 transition-colors"
   704→            >
   705→              <span style={{ color: CATEGORY_DOT_COLORS[t.category] }}>
   706→                <Icon className="h-3.5 w-3.5" />
   707→              </span>
   708→              <span className="font-medium">{t.name}</span>
   709→            </button>
   710→          );
   711→        })}
   712→      </div>
   713→    </div>
   714→  );
