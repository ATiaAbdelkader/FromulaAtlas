  1617→  1617→    var x2 = TP.x, y2 = TP.y;

  1622→  1622→    var wClay = 1 - wSand - wSilt;

  1624→  1624→    wSilt = Math.max(0, wSilt);
  1625→  1625→    wClay = Math.max(0, wClay);
  1626→  1626→    var s = wSand + wSilt + wClay;
  1627→  1627→    if (s < 1e-9) return { sand: 33.3, silt: 33.3, clay: 33.4 };
