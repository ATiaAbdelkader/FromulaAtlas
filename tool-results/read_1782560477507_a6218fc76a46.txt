   332→   332→    function calculateSaturationAbsoluteHumidity(tempC) {
   333→   333→      return 5.018 + (0.32321 * tempC) + (0.0081847 * tempC * tempC) + (0.00031243 * tempC * tempC * tempC);
   334→   334→    }

   336→   336→    function calculateVPDSimple(airTemp, humidity) {
   337→   337→      const es = 0.6108 * Math.exp((17.27 * airTemp) / (airTemp + 237.3));
   338→   338→      const ea = es * (humidity / 100);
   339→   339→      const vpd = es - ea;
   340→   340→      const hd = calculateSaturationAbsoluteHumidity(airTemp) * ((100 - humidity) / 100);
   341→   341→      return { vpd: parseFloat(vpd.toFixed(2)), hd: parseFloat(hd.toFixed(2)) };
   342→   342→    }

   344→   344→    function calculateLeafTempFromRadiation(airTemp, solarRadiation) {
   345→   345→      const leafTemp = solarRadiation > 200
   346→   346→        ? airTemp + (((solarRadiation - 200) * 0.6) / 100)
   347→   347→        : airTemp;
   348→   348→      return parseFloat(leafTemp.toFixed(1));
   349→   349→    }

   351→   351→    function calculateVPDAdvanced(airTemp, airHumidity, leafTemp) {
   352→   352→      const esLeaf = 0.6108 * Math.exp((17.27 * leafTemp) / (leafTemp + 237.3));
   353→   353→      const esAir = 0.6108 * Math.exp((17.27 * airTemp) / (airTemp + 237.3));
   354→   354→      const ea = esAir * (airHumidity / 100);
   355→   355→      const vpd = esLeaf - ea;
   356→   356→      const hd = calculateSaturationAbsoluteHumidity(leafTemp) -
   357→   357→        (calculateSaturationAbsoluteHumidity(airTemp) * (airHumidity / 100));
   358→   358→      return { vpd: parseFloat(vpd.toFixed(2)), hd: parseFloat(hd.toFixed(2)) };
   359→   359→    }

   482→   482→      var uv = cur.uv_index;
   483→   483→      var uvNum = (uv != null && Number.isFinite(Number(uv))) ? Number(uv) : null;
   484→   484→      return {
   485→   485→        temperature: cur.temperature_2m,
   486→   486→        humidity: cur.relative_humidity_2m,
   487→   487→        shortwaveRadiation: radNum,
   488→   488→        uvIndex: uvNum
