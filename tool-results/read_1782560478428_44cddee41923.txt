   689→    /**
   690→     * Tramos mg/L como CaCO₃ alineados con referencia tipo USGS: dura 121–180,

   693→    function hardnessClassByPpm(ppm) {
   694→      if (ppm < 60) return 'Blanda';
   695→      if (ppm < 120) return 'Moderadamente dura';
   696→      if (ppm < 180) return 'Dura';
   697→      return 'Muy dura';
   698→    }

   700→    /**
   701→     * Sincroniza las 4 casillas "otras" a partir de ppm; NO escribe en skipKey.
   702→     * skipKey: 'ppm'|'meq'|'dh'|'eh'|'fh'|null
   703→     */
   704→    function syncHardnessFromPpm(ppm, skipKey) {
   705→      const meq = ppm / EQ_CACO3;
   706→      const dHv = ppm / PPM_PER_DH;
   707→      const eHv = ppm / PPM_PER_EH;
   708→      const fHv = ppm / PPM_PER_FH;
   709→      if (skipKey !== 'ppm') el.ppm.value = f(ppm, 2);
   710→      if (skipKey !== 'meq') el.meq.value = f(meq, 2);
   711→      if (skipKey !== 'dh') el.dh.value = f(dHv, 2);
   712→      if (skipKey !== 'eh') el.eh.value = f(eHv, 2);
   713→      if (skipKey !== 'fh') el.fh.value = f(fHv, 2);
   714→
   715→      const cls = hardnessClassByPpm(ppm);
   716→      el.hClass.textContent = `Clasificación: ${cls}`;
   717→      el.hSummary.innerHTML = `
   718→        <strong>Lectura rápida:</strong> ${cls} · ${f(ppm, 1)} ppm como CaCO₃.<br>
   719→        <span class="muted">Equivalente estimado: ${f(meq, 2)} meq/L · ${f(dHv, 2)} °dH · ${f(eHv, 2)} °e · ${f(fHv, 2)} °fH.</span>
   720→      `;
   721→    }

   724→      const v = numOrZeroForDisplay(raw);

   728→      if (key === 'eh') return v * PPM_PER_EH;
   729→      if (key === 'fh') return v * PPM_PER_FH;

   761→    function updateAcidCalc() {
   762→      const acid = acids.find(a => a.id === el.acid.value) || acids[0];
   763→      const meqPerMl = parseFloat(acid && acid.meqPerMl) || 0;
   764→      const rho = parseFloat(acid && acid.densityKgL) || 0;
   765→
   766→      const vol = getAcidVolume();
   767→      const volStr = vol.raw;
   768→      const volL = vol.volL;
   769→      const volM3 = vol.volM3;
   770→
   771→      if (volStr === '' || Number.isNaN(volL)) {
   772→        setAcidPlaceholderUI();
   773→        return;
   774→      }
   775→
   776→      if (el.volM3Equiv) {
   777→        el.volM3Equiv.innerHTML = '<span class="muted">Volumen usado: <strong>' + f(vol.inputValue, 3) + ' ' + (vol.unit === 'm3' ? 'm³' : 'L') + '</strong> = <strong>' + f(volM3, 3) + '</strong> m³ = <strong>' + f(volL, 2) + '</strong> L.</span>';
   778→      }
   779→
   780→      const hco3 = n(el.hco3.value);
   781→      const co3 = n(el.co3.value);
   782→      const residual = n(el.residual.value);
   783→      const sourceMeq = hco3 + co3;
   784→      const needMeq = Math.max(0, sourceMeq - residual);
   785→      // Igual que Análisis de agua: meq de ácido por m³ de agua = (meq/L) × 1000; mL/m³ = meq/m³ ÷ meq/mL del ácido.
   786→      const meqPerM3 = needMeq * 1000;
   787→      const mlPerM3 = meqPerMl > 0 ? meqPerM3 / meqPerMl : 0;
   788→      const totalMl = mlPerM3 * volM3;
   789→      const totalLAcid = totalMl / 1000;
   790→      const kgPerM3Water = (mlPerM3 / 1000) * rho;
   791→      const kgTotal = totalLAcid * rho;
   792→
   793→      el.acidResult.innerHTML = `
   794→        <strong>Resultado con ${acid.name}:</strong><br>
   795→        Meq/L a neutralizar = (${f(hco3, 2)} + ${f(co3, 2)}) − ${f(residual, 2)} = <strong>${f(needMeq, 2)} meq/L</strong><br>
   796→        Dosis = <strong>${f(mlPerM3, 2)} mL/m³</strong> de ácido <span class="muted">(${f(kgPerM3Water, 3)} kg de ácido / m³ de agua, aprox.)</span><br>
   797→        Para <strong>${f(vol.inputValue, vol.unit === 'm3' ? 3 : 2)} ${vol.unit === 'm3' ? 'm³' : 'L'}</strong> de agua (${f(volM3, 3)} m³ / ${f(volL, 2)} L): <strong>${f(totalLAcid, 3)} L</strong> de ácido (${f(totalMl, 1)} mL) · aprox. <strong>${f(kgTotal, 3)} kg</strong>
   798→      `;
   799→    }
