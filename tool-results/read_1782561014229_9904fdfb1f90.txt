   826→  function normalizeState(raw) {
   827→    if (!raw || typeof raw !== 'object') return null;
   828→    if (!Array.isArray(raw.nutrients) || !Array.isArray(raw.stages) || !raw.pct || typeof raw.pct !== 'object') return null;
   829→    var nList = raw.nutrients
   830→      .filter(function (n) { return n && typeof n.id === 'string' && typeof n.label === 'string'; })
   831→      .map(function (n) {
   832→        return {
   833→          id: n.id,
   834→          label: n.label,
   835→          total: parseFloat(n.total) || 0,
   836→          optional: !!n.optional
   837→        };
   838→      });
   839→    if (!nList.length) return null;
   840→    var st = raw.stages.map(function (s, i) {
   841→      var txt = String(s || '').trim();
   842→      return txt || ('Etapa ' + (i + 1));
   843→    });
   844→    if (!st.length) return null;
   845→    var pctMap = {};
   846→    nList.forEach(function (n) {
   847→      var arr = Array.isArray(raw.pct[n.id]) ? raw.pct[n.id].slice(0, st.length) : [];
   848→      while (arr.length < st.length) arr.push(0);
   849→      pctMap[n.id] = arr.map(function (v) { return round1(parseFloat(v) || 0); });
   850→    });
   851→    return {
   852→      version: raw.version || toolStateVersion,
   853→      updatedAt: Number(raw.updatedAt) || Date.now(),
   854→      nutrients: nList,
   855→      optionalQueue: Array.isArray(raw.optionalQueue) ? raw.optionalQueue : OPTIONAL_QUEUE.filter(function (o) {
   856→        return !nList.some(function (n) { return n.id === o.id; });
   857→      }),
   858→      stages: st,
   859→      pct: pctMap
   860→    };
   861→  }

   969→  function ensurePct() {
   970→    nutrients.forEach(function (n) {
   971→      if (!pct[n.id]) {
   972→        var eq = equalSplit(stages.length);
   973→        pct[n.id] = eq;
   974→      } else if (pct[n.id].length !== stages.length) {
   975→        resizePctArray(n.id, stages.length);
   976→      }
   977→    });

   982→  function equalSplit(n) {
   983→    if (n <= 0) return [];
   984→    var arr = [];
   985→    var s = 0;
   986→    var q = 100 / n;
   987→    for (var i = 0; i < n - 1; i++) {
   988→      var v = round1(q);
   989→      arr.push(v);
   990→      s += v;
   991→    }
   992→    arr.push(round1(100 - s));
   993→    return arr;
   994→  }
   995→  function resizePctArray(id, newLen) {
   996→    var old = pct[id] || [];
   997→    if (old.length === newLen) return;
   998→    while (old.length < newLen) {
   999→      if (!old.length) {
  1000→        old = equalSplit(newLen);
  1001→        pct[id] = old;
  1002→        return;
  1003→      }
  1004→      var last = parseFloat(old[old.length - 1]) || 0;
  1005→      var a = round1(last / 2);
  1006→      var b = round1(last - a);
  1007→      old[old.length - 1] = a;
  1008→      old.push(b);
  1009→      var diff = 100 - sumPct(id);
  1010→      if (Math.abs(diff) > 0.05) {
  1011→        old[old.length - 1] = round1((parseFloat(old[old.length - 1]) || 0) + diff);
  1012→      }
  1013→    }
  1014→    while (old.length > newLen) {
  1015→      var rm = parseFloat(old[old.length - 1]) || 0;
  1016→      old.pop();
  1017→      if (old.length) {
  1018→        old[old.length - 1] = round1((parseFloat(old[old.length - 1]) || 0) + rm);
  1019→      }
  1020→    }
  1021→    pct[id] = old;
  1022→  }
  1023→  function round1(x) { return Math.round(x * 10) / 10; }

  1025→  /** kg/ha legible: pocos decimales y sin basura de float */
  1026→  function formatKgHa(v) {
  1027→    var x = Number(v);
  1028→    if (!isFinite(x)) return '0';
  1029→    x = Math.round(x * 1e6) / 1e6;
  1030→    var a = Math.abs(x);
  1031→    if (a >= 100) return (Math.round(x * 10) / 10).toFixed(1);
  1032→    if (a >= 10) return (Math.round(x * 100) / 100).toFixed(2);
  1033→    if (a >= 1) return (Math.round(x * 100) / 100).toFixed(2);
  1034→    if (a >= 0.1) return (Math.round(x * 1000) / 1000).toFixed(3);
  1035→    if (a >= 0.01) return (Math.round(x * 10000) / 10000).toFixed(4);
  1036→    return (Math.round(x * 1e6) / 1e6).toFixed(6).replace(/\.?0+$/, '') || '0';
  1037→  }
