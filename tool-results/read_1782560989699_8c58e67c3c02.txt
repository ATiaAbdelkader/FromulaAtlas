   309→   309→          const term = terms[i + j];
   310→   310→          const m = motif[j];
   311→   311→          if (!term || term.type !== 'element' || term.symbol !== m.symbol || Math.abs(term.count - m.count) > 1e-9) {
   312→   312→            ok = false;
   313→   313→            break;
   314→   314→          }
   315→   315→        }
   316→   316→        if (ok) total += mult;
   317→   317→      }
   318→   318→      terms.forEach(t => {
   319→   319→        if (t.type === 'group') total += countMotifInTerms(t.terms, mult * t.count, motif);
   320→   320→      });
   321→   321→      return total;
   322→   322→    }

   324→   324→    function parseCompound(rawFormula) {
   325→   325→      const normalized = normalizeFormula(rawFormula);
   326→   326→      if (!normalized) throw new Error('Escribe una fórmula.');
   327→   327→      const parts = normalized.split(/[·•.]/).filter(Boolean);
   328→   328→      if (!parts.length) throw new Error('Fórmula no válida.');
   329→   329→      const segments = [];
   330→   330→      parts.forEach(part => {
   331→   331→        const m = part.match(/^(\d+(?:\.\d+)?)(.*)$/);
   332→   332→        let coef = 1;
   333→   333→        let body = part;
   334→   334→        if (m && m[2]) {
   335→   335→          coef = Number(m[1]);
   336→   336→          body = m[2];
   337→   337→        }
   338→   338→        const parsed = parseTerms(body, 0, null);
   339→   339→        if (parsed.index !== body.length) throw new Error('Error al parsear la fórmula.');
   340→   340→        segments.push({ coef, terms: parsed.terms });
   341→   341→      });
   342→   342→      const atomCounts = {};
   343→   343→      segments.forEach(seg => collectElements(seg.terms, seg.coef, atomCounts));
   344→   344→      let mw = 0;
   345→   345→      Object.keys(atomCounts).forEach(sym => { mw += atomCounts[sym] * ELEM[sym]; });
   346→   346→      if (!(mw > 0)) throw new Error('No se pudo calcular el PM.');
   347→   347→      const no3Count = segments.reduce((s, seg) =>
   348→   348→        s + countMotifInTerms(seg.terms, seg.coef, [{ symbol: 'N', count: 1 }, { symbol: 'O', count: 3 }]), 0);
   349→   349→      const nh4Count = segments.reduce((s, seg) =>
   350→   350→        s + countMotifInTerms(seg.terms, seg.coef, [{ symbol: 'N', count: 1 }, { symbol: 'H', count: 4 }]), 0);
   351→   351→      return { normalized, atomCounts, mw, no3Count, nh4Count };
   352→   352→    }

   354→   354→    function pctElem(atomCounts, sym, mw) {
   355→   355→      return ((atomCounts[sym] || 0) * ELEM[sym] / mw) * 100;
   356→   356→    }

   358→   358→    function compFromFormula(formula) {
   359→   359→      const parsed = parseCompound(formula);
   360→   360→      const c = parsed.atomCounts;
   361→   361→      const mw = parsed.mw;
   362→   362→      const out = zeroComp();
   363→   363→      out.N = pctElem(c, 'N', mw);
   364→   364→      out.P = pctElem(c, 'P', mw);
   365→   365→      out.K = pctElem(c, 'K', mw);
   366→   366→      out.Ca = pctElem(c, 'Ca', mw);
   367→   367→      out.Mg = pctElem(c, 'Mg', mw);
   368→   368→      out.S = pctElem(c, 'S', mw);
   369→   369→      out.Si = pctElem(c, 'Si', mw);
   370→   370→      out.C = pctElem(c, 'C', mw);
   371→   371→      out.H = pctElem(c, 'H', mw);
   372→   372→      out.O = pctElem(c, 'O', mw);
   373→   373→      out.Zn = pctElem(c, 'Zn', mw);
   374→   374→      out.Fe = pctElem(c, 'Fe', mw);
   375→   375→      out.Mn = pctElem(c, 'Mn', mw);
   376→   376→      out.B = pctElem(c, 'B', mw);
   377→   377→      out.Cu = pctElem(c, 'Cu', mw);
   378→   378→      out.Mo = pctElem(c, 'Mo', mw);
   379→   379→
   380→   380→      const nAtoms = c.N || 0;
   381→   381→      if (nAtoms > 0) {
   382→   382→        let no3Atoms = Math.max(0, parsed.no3Count);
   383→   383→        let nh4Atoms = Math.max(0, parsed.nh4Count);
   384→   384→        const forms = no3Atoms + nh4Atoms;
   385→   385→        if (forms > nAtoms && forms > 0) {
   386→   386→          const k = nAtoms / forms;
   387→   387→          no3Atoms *= k;
   388→   388→          nh4Atoms *= k;
   389→   389→        }
   390→   390→        out.N_NO3 = forms > 0 ? out.N * (no3Atoms / nAtoms) : 0;
   391→   391→        out.N_NH4 = forms > 0 ? out.N * (nh4Atoms / nAtoms) : 0;
   392→   392→      }
   393→   393→      out.P2O5 = out.P * OX.P2O5;
   394→   394→      out.K2O = out.K * OX.K2O;
   395→   395→      out.CaO = out.Ca * OX.CaO;
   396→   396→      out.MgO = out.Mg * OX.MgO;
   397→   397→      out.SiO2 = out.Si * OX.SiO2;
   398→   398→      return { comp: out, mw };
   399→   399→    }

   698→   698→    function recalcBlend() {
   699→   699→      othersPct = num(document.getElementById('othersPctInput').value);
   700→   700→      const totals = zeroComp();
   701→   701→      let sumMat = 0;
   702→   702→      blendRows.forEach(function(r) {
   703→   703→        const p = Math.max(0, num(r.pct));
   704→   704→        sumMat += p;
   705→   705→        const info = resolveMaterial(r.material);
   706→   706→        if (info.ok) {
   707→   707→          COLS.forEach(function(k) {
   708→   708→            totals[k] += (p / 100) * num(info.comp[k]);
   709→   709→          });
   710→   710→        }
   711→   711→      });
   712→   712→      const totalPct = sumMat + othersPct;
   713→   713→      const box = document.getElementById('blendValidation');
   714→   714→      const totalBox = document.getElementById('blendTotalBox');
   715→   715→      const nutDiv = document.getElementById('blendTotalNutrients');
   716→   716→      if (Math.abs(totalPct - 100) <= 0.02 && blendRows.some(function(r) { return String(r.material).trim(); })) {
   717→   717→        box.textContent = 'Suma: ' + totalPct.toFixed(2) + '% — OK.';
   718→   718→        box.style.color = '#166534';
   719→   719→        totalBox.style.display = 'block';
   720→   720→        var labels = { N: 'N', P: 'P', K: 'K', Ca: 'Ca', Mg: 'Mg', S: 'S', Si: 'Si', C: 'C', H: 'H', O: 'O', Zn: 'Zn', Fe: 'Fe', Mn: 'Mn', B: 'B', Cu: 'Cu', Mo: 'Mo',
   721→   721→          N_NO3: 'N-NO₃', N_NH4: 'N-NH₄', P2O5: 'P₂O₅', K2O: 'K₂O', CaO: 'CaO', MgO: 'MgO', SiO2: 'SiO₂' };
   722→   722→        var elemOrder = ['N', 'N_NO3', 'N_NH4', 'P', 'K', 'Ca', 'Mg', 'S', 'Si', 'Zn', 'Fe', 'Mn', 'B', 'Cu', 'Mo', 'C', 'H', 'O'];
   723→   723→        var oxOrder = ['P2O5', 'K2O', 'CaO', 'MgO', 'SiO2'];
   724→   724→        var lineElem = [];
   725→   725→        elemOrder.forEach(function(k) {
   726→   726→          if (totals[k] > 1e-4) lineElem.push('<strong>' + labels[k] + '</strong>: ' + totals[k].toFixed(2) + ' %');
   727→   727→        });
   728→   728→        var lineOx = [];
   729→   729→        oxOrder.forEach(function(k) {
   730→   730→          if (totals[k] > 1e-4) lineOx.push('<strong>' + labels[k] + '</strong>: ' + totals[k].toFixed(2) + ' %');
   731→   731→        });
   732→   732→        nutDiv.innerHTML =
   733→   733→          '<p class="blend-total-label">% ponderado — forma elemental (todo lo que aporta)</p>' +
   734→   734→          '<p class="blend-total-line">' + (lineElem.length ? lineElem.join(' · ') : '—') + '</p>' +
   735→   735→          '<div class="blend-detail-card">' +
   736→   736→          '<div class="blend-detail-card-title">Equivalente en óxido (P₂O₅, K₂O, CaO, MgO, SiO₂)</div>' +

   738→   738→          '</div>';

   741→   741→          ? ('Suma: ' + totalPct.toFixed(2) + '% — Ajusta a 100% (ingredientes + Otros).')

   743→   743→        box.style.color = '#b91c1c';
   744→   744→        totalBox.style.display = 'none';
   745→   745→      }
